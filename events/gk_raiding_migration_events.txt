# MIGRATION — safe to delete once all saves have loaded with this version
# 1-2: bucket rate migration (Phase 1)
# 10-11: robo-liberator civic deprecation (Phase 3)
namespace = gk_raiding_migration

# SP entry point — fires the migration on global_event_country
event = {
	id = gk_raiding_migration.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { has_global_flag = gk_migration_bucket_rate_done }
	}

	immediate = {
		event_target:global_event_country = {
			country_event = { id = gk_raiding_migration.2 }
		}
	}
}

# Migration body — runs on global_event_country
country_event = {
	id = gk_raiding_migration.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { has_global_flag = gk_migration_bucket_rate_done }
	}

	immediate = {
		set_global_flag = gk_migration_bucket_rate_done
		# Patch missing vars and reset bucket on old trackers
		every_owned_leader = {
			limit = { has_leader_flag = gk_raid_tracker }
			if = {
				limit = { NOT = { is_variable_set = gk_escalation_tier } }
				set_variable = { which = gk_escalation_tier value = 0 }
			}
			set_variable = { which = gk_bucket_pops_taken value = 0 }
		}
		# Register global event targets for active wars
		every_country = {
			limit = { is_at_war = yes }
			every_war = {
				save_event_target_as = gk_current_war
				# Attacker-side tracker
				attacker = { save_event_target_as = gk_cap_raider }
				defender = { save_event_target_as = gk_raidee }
				if = {
					limit = {
						event_target:global_event_country = {
							any_owned_leader = {
								has_leader_flag = gk_raid_tracker
								has_leader_flag = gk_raid_tracker_attacker_@event_target:gk_cap_raider
								has_leader_flag = gk_raid_tracker_defender_@event_target:gk_raidee
							}
						}
					}
					event_target:global_event_country = {
						random_owned_leader = {
							limit = {
								has_leader_flag = gk_raid_tracker
								has_leader_flag = gk_raid_tracker_attacker_@event_target:gk_cap_raider
								has_leader_flag = gk_raid_tracker_defender_@event_target:gk_raidee
							}
							save_global_event_target_as = gk_raid_tracker_atk_@event_target:gk_current_war
						}
					}
				}
				# Defender-side tracker
				defender = { save_event_target_as = gk_cap_raider }
				attacker = { save_event_target_as = gk_raidee }
				if = {
					limit = {
						event_target:global_event_country = {
							any_owned_leader = {
								has_leader_flag = gk_raid_tracker
								has_leader_flag = gk_raid_tracker_attacker_@event_target:gk_cap_raider
								has_leader_flag = gk_raid_tracker_defender_@event_target:gk_raidee
							}
						}
					}
					event_target:global_event_country = {
						random_owned_leader = {
							limit = {
								has_leader_flag = gk_raid_tracker
								has_leader_flag = gk_raid_tracker_attacker_@event_target:gk_cap_raider
								has_leader_flag = gk_raid_tracker_defender_@event_target:gk_raidee
							}
							save_global_event_target_as = gk_raid_tracker_def_@event_target:gk_current_war
						}
					}
				}
			}
		}
	}
}
