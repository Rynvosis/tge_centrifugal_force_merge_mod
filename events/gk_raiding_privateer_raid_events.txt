# 201 — Gatekeeper: clicking a privateer fleet opens the parent enclave's diplomacy
# This = player country, From = temp raider country (gk_privateer_raiders)
country_event = {
	id = gk_raiding_privateers.201
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		from = {
			is_country_type = gk_privateer_raiders
		}
		NOT = {
			has_active_event = {
				gk_raiding_privateers.205
			}
		}
	}

	immediate = {
		# Find the parent enclave via the raider's flag
		random_country = {
			limit = {
				is_country_type = gk_privateer_enclave
				from = { has_country_flag = gk_privateer_raider_of_@prev }
			}
			save_event_target_as = gk_privateer_enclave_country
		}

		if = {
			limit = { exists = event_target:gk_privateer_enclave_country }
			country_event = { id = gk_raiding_privateers.205 }
		}
	}
}

# 211 — Target + fleet selection
# This = enclave country
country_event = {
	id = gk_raiding_privateers.211
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		# Select fleet
		random_owned_fleet = {
			limit = {
				gk_is_eligible_raid_fleet = yes
			}
			save_event_target_as = gk_raid_fleet
			set_fleet_flag = gk_fleet_locked
		}

		# Select target
		random_country = {
			limit = {
				gk_is_valid_raid_target = yes
			}
			save_event_target_as = gk_raid_target
		}

		if = {
			limit = {
				exists = event_target:gk_raid_fleet
				exists = event_target:gk_raid_target
			}
			set_country_flag = gk_auto_raid_ongoing

			# Send tribute demand to target
			event_target:gk_raid_target = {
				country_event = { id = gk_raiding_privateers.400 }
			}
		}
		else = {
			# Couldn't find fleet or target; unlock if we locked
			if = {
				limit = { exists = event_target:gk_raid_fleet }
				event_target:gk_raid_fleet = {
					remove_fleet_flag = gk_fleet_locked
				}
			}
		}
	}
}

# 212 — Fleet dispatch: create temp raider country, transfer fleet, send to target
# This = enclave country
# event_target:gk_raid_fleet = locked fleet
# event_target:gk_raid_target = target empire
country_event = {
	id = gk_raiding_privateers.212
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		save_event_target_as = gk_privateer_enclave_country

		# Create temp raider country
		create_country = {
			name = root
			type = gk_privateer_raiders
			species = root.species
			flag = root.flag
			effect = {
				save_event_target_as = gk_privateer_raider
				set_country_flag = gk_privateer_raider_of_@root
				set_country_flag = gk_privateer_targeting_@event_target:gk_raid_target
				establish_communications_no_message = event_target:gk_raid_target
			}
		}

		# Establish comms with all countries (prevents engine garbage-collection)
		every_country = {
			establish_communications_no_message = event_target:gk_privateer_raider
		}

		# Set raider hostile to target only
		event_target:gk_privateer_raider = {
			set_faction_hostility = {
				target = event_target:gk_raid_target
				set_hostile = yes
				set_neutral = no
				set_friendly = no
			}
		}

		# Set enclave friendly to raider
		set_faction_hostility = {
			target = event_target:gk_privateer_raider
			set_hostile = no
			set_neutral = no
			set_friendly = yes
		}

		# Store relationship flags on enclave
		set_country_flag = gk_has_active_privateer_raider

		# Pick a target planet
		event_target:gk_raid_target = {
			random_owned_planet = {
				limit = {
					is_colony = yes
					count_owned_pop_amount = { count >= 1 }
				}
				save_event_target_as = gk_raid_planet
			}
		}

		# Transfer fleet to temp raider and send to target
		event_target:gk_raid_fleet = {
			set_owner = event_target:gk_privateer_raider
			set_fleet_bombardment_stance = indiscriminate
			set_fleet_stance = aggressive
			set_aggro_range_measure_from = self
			set_aggro_range = 150

			# Move to target planet
			if = {
				limit = { exists = event_target:gk_raid_planet }
				auto_move_to_planet = {
					target = event_target:gk_raid_planet
					clear_auto_move_on_arrival = no
				}
			}

			# Safety fallback: withdraw after 5 years if devastation trigger never fires
			fleet_event = { id = gk_raiding_privateers.220 days = 1800 }
		}
	}
}

# 215 — Orphaned raid cleanup (monthly)
# Detects if a raid is ongoing but the temp raider country is gone
# This = enclave country
country_event = {
	id = gk_raiding_privateers.215
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_country_type = gk_privateer_enclave
		has_country_flag = gk_auto_raid_ongoing
		NOT = {
			any_country = {
				is_country_type = gk_privateer_raiders
				has_country_flag = gk_privateer_raider_of_@root
			}
		}
	}

	immediate = {
		# No temp raider exists; raid failed or was cleaned up externally
		remove_country_flag = gk_auto_raid_ongoing
		remove_country_flag = gk_has_active_privateer_raider
		set_timed_country_flag = {
			flag = gk_auto_raid_cooldown
			days = 360
		}

		# Unlock any lingering locked fleets
		every_owned_fleet = {
			limit = { has_fleet_flag = gk_fleet_locked }
			remove_fleet_flag = gk_fleet_locked
		}
	}
}

# 230 — Privateer bombardment pillage: accumulate loot on receipt
# This = Planet
# From = Bombarder country (gk_privateer_raiders)
planet_event = {
	id = gk_raiding_privateers.230
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = owner
		is_colony = yes
		exists = from
		from = {
			is_country_type = gk_privateer_raiders
		}
	}

	immediate = {
		from = {
			save_event_target_as = gk_raider
		}
		save_event_target_as = gk_raiding_target

		# Use a per-planet receipt for the temp raider
		gk_get_or_create_loot_receipt = yes

		# Track fresh devastation on the receipt
		event_target:gk_loot_receipt = {
			change_variable = {
				which = gk_pillage_damage_bank
				value = root.local_devastation
			}
		}

		# Only resolve packets when we have crossed a full devastation threshold
		if = {
			limit = {
				event_target:gk_loot_receipt = {
					check_variable = {
						which = gk_pillage_damage_bank
						value >= @gk_pillage_devastation_per_packet
					}
				}
			}
			set_variable = {
				which = gk_pillage_packet_count
				value = event_target:gk_loot_receipt.gk_pillage_damage_bank
			}
			divide_variable = {
				which = gk_pillage_packet_count
				value = @gk_pillage_devastation_per_packet
			}
			floor_variable = gk_pillage_packet_count

			set_variable = {
				which = gk_pillage_damage_to_consume
				value = gk_pillage_packet_count
			}
			multiply_variable = {
				which = gk_pillage_damage_to_consume
				value = @gk_pillage_devastation_per_packet
			}
			event_target:gk_loot_receipt = {
				subtract_variable = {
					which = gk_pillage_damage_bank
					value = prev.gk_pillage_damage_to_consume
				}
			}

			# Use planet as temp storage, calculate loot, then accumulate on receipt
			save_event_target_as = gk_storage_scope
			gk_init_loot_variables = yes
			set_variable = {
				which = gk_total_loot_packets
				value = gk_pillage_packet_count
			}
			gk_calculate_base_loot = yes
			gk_transform_loot = yes

			# Move transformed loot from planet to receipt
			gk_accumulate_loot_on_receipt = yes

			# Clean up planet temp vars
			gk_clear_loot_variables = yes
			clear_variable = gk_pillage_packet_count
			clear_variable = gk_pillage_damage_to_consume
		}
	}
}

# 231 — Devastation-based withdrawal trigger
# Fires alongside 230 on_planet_bombarded; schedules withdrawal once target is sufficiently damaged
# This = Planet, From = Bombarder country (gk_privateer_raiders)
planet_event = {
	id = gk_raiding_privateers.231
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = owner
		is_colony = yes
		exists = from
		from = {
			is_country_type = gk_privateer_raiders
			NOT = { has_country_flag = gk_privateer_raid_accomplished }
		}
		# 51 ensures the 50-devastation loot packet fires before withdrawal
		planet_devastation >= 51
	}

	immediate = {
		from = {
			set_country_flag = gk_privateer_raid_accomplished

			# Schedule withdrawal on the raider's fleet (30–60 days grace period for final loot)
			random_owned_fleet = {
				fleet_event = { id = gk_raiding_privateers.220 days = 30 random = 30 }
			}
		}
	}
}

# 341 — Hidden dispatch for commissioned raid
# This = enclave country (fired from patron's event 340)
# gk_commissioned_raid_target flag on target country
# gk_fleet_locked flag on one fleet
country_event = {
	id = gk_raiding_privateers.341
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		save_event_target_as = gk_privateer_enclave_country

		# Find the commissioned target
		random_country = {
			limit = { has_country_flag = gk_commissioned_raid_target }
			save_event_target_as = gk_raid_target
			remove_country_flag = gk_commissioned_raid_target
		}

		# Find and lock an eligible fleet
		random_owned_fleet = {
			limit = { gk_is_eligible_raid_fleet = yes }
			save_event_target_as = gk_raid_fleet
			set_fleet_flag = gk_fleet_locked
		}

		if = {
			limit = {
				exists = event_target:gk_raid_target
				exists = event_target:gk_raid_fleet
			}
			set_country_flag = gk_auto_raid_ongoing
			# Track who commissioned this raid (for cancel raid feature)
			set_country_flag = gk_raid_commissioned
			if = {
				limit = { exists = from }
				set_country_flag = gk_raid_commissioned_by_@from
			}
			country_event = { id = gk_raiding_privateers.212 }
		}
		else = {
			# Fallback: clean up stale locks (target or fleet vanished between events)
			every_owned_fleet = {
				limit = { has_fleet_flag = gk_fleet_locked }
				remove_fleet_flag = gk_fleet_locked
			}
		}
	}
}

		set_country_flag = gk_tribute_responded

		# Find patron and give them the tribute energy
		event_target:gk_privateer_enclave_country = {
			gk_get_privateer_station_context = yes

			# Unlock fleet
			if = {
				limit = { exists = event_target:gk_raid_fleet }
				event_target:gk_raid_fleet = {
					remove_fleet_flag = gk_fleet_locked
				}
			}

			# Clear raid state
			remove_country_flag = gk_auto_raid_ongoing

			# Set 2-year truce (can't raid same target)
			root = {
				set_timed_country_flag = {
					flag = gk_raided_by_@event_target:gk_privateer_enclave_country
					days = 720
				}
			}

			# Cooldown before next autonomous raid
			set_timed_country_flag = {
				flag = gk_auto_raid_cooldown
				days = 540
			}
		}

		# Transfer tribute to patron
		if = {
			limit = { exists = event_target:gk_privateer_patron }
			event_target:gk_privateer_patron = {
				# Reimburse the energy that was paid as tribute
				if = {
					limit = {
						root = { has_monthly_income = { resource = energy value < 100 } }
					}
					add_resource = { energy = 500 }
				}
				else_if = {
					limit = {
						root = { has_monthly_income = { resource = energy value < 200 } }
					}
					add_resource = { energy = 1000 }
				}
				else_if = {
					limit = {
						root = { has_monthly_income = { resource = energy value < 400 } }
					}
					add_resource = { energy = 2000 }
				}
				else = {
					add_resource = { energy = 3000 }
				}
			}
		}
	}
}

# 615 — Monthly: return idle fleets to home system
# This = enclave country (from on_monthly_pulse_country)
country_event = {
	id = gk_raiding_privateers.615
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_country_type = gk_privateer_enclave
		NOT = { has_country_flag = gk_auto_raid_ongoing }
	}

	immediate = {
		# Find station system, star, and patron
		gk_get_privateer_station_context = yes

		# Send any idle non-station fleet back home
		every_owned_fleet = {
			limit = {
				is_fleet_idle = yes
				NOT = { has_fleet_flag = gk_fleet_locked }
				NOT = {
					any_owned_ship = {
						is_ship_size = gk_privateer_station
					}
				}
				OR = {
					NOT = { exists = event_target:gk_privateer_enclave_system }
					NOT = {
						solar_system = {
							is_same_value = event_target:gk_privateer_enclave_system
						}
					}
				}
			}
			if = {
				limit = { exists = event_target:gk_enclave_star }
				queue_actions = {
					move_to = event_target:gk_enclave_star
				}
			}
		}
	}
}

# 620 — Monthly global pulse: orphaned raider country failsafe
# Finds any gk_privateer_raiders country with no ships or no matching parent enclave
# This = n/a (global scope, from on_monthly_pulse)
event = {
	id = gk_raiding_privateers.620
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		any_country = {
			is_country_type = gk_privateer_raiders
			OR = {
				# No ships left
				NOT = {
					any_owned_fleet = {
						num_ships > 0
					}
				}
				# No matching parent enclave
				NOT = {
					any_country = {
						is_country_type = gk_privateer_enclave
						prev = { has_country_flag = gk_privateer_raider_of_@prev }
					}
				}
			}
		}
	}

	immediate = {
		every_country = {
			limit = {
				is_country_type = gk_privateer_raiders
				OR = {
					NOT = {
						any_owned_fleet = {
							num_ships > 0
						}
					}
					NOT = {
						any_country = {
							is_country_type = gk_privateer_enclave
							prev = { has_country_flag = gk_privateer_raider_of_@prev }
						}
					}
				}
			}
			save_event_target_as = gk_privateer_raider

			# Inline cleanup (can't use gk_cleanup_privateer_raid: scope is raider, not enclave)
			# Clean up any lingering receipts
			event_target:global_event_country = {
				every_owned_leader = {
					limit = {
						has_leader_flag = gk_loot_receipt_to_@event_target:gk_privateer_raider
					}
					kill_leader = { show_notification = no }
				}
			}

			# Delete fleets and destroy country
			every_owned_fleet = {
				delete_fleet = this
			}
			destroy_country = yes
		}
	}
}
