namespace = rp_util

# Multiplayer detection — fired on game start
event = {
	id = rp_util.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_multiplayer = yes
	}

	immediate = {
		set_global_flag = rp_mp_game
	}
}

# Enable Galactic Community — fire via console: event rp_util.2
event = {
	id = rp_util.2
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		set_global_flag = rp_galcom_allowed
		# Kick off vanilla GC formation
		random_country = {
			limit = { is_country_type = global_event }
			country_event = { id = galcom.1 }
		}
	}
}
