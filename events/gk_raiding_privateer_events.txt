# Merc -> Privateer: execute conversion — hidden
# This = enclave country (was enclave_mercenary, becomes gk_privateer_enclave)
# Fired from overwritten enclave.205
country_event = {
	id = gk_raiding_privateers.112
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		save_event_target_as = gk_privateer_enclave_country

		# Find patron before conversion (while station is still merc type)
		random_owned_fleet = {
			limit = { is_ship_size = enclave_mercenary_station }
			solar_system = {
				if = {
					limit = { exists = space_owner }
					space_owner = {
						save_event_target_as = gk_privateer_patron
					}
				}
				save_event_target_as = gk_privateer_enclave_system
			}
			# Remember station location
			random_owned_ship = {
				save_event_target_as = gk_old_station_ship
			}
		}

		# Change country type and civic
		set_country_type = gk_privateer_enclave
		force_remove_civic = civic_mercenary_enclave
		force_add_civic = civic_gk_privateer_enclave
		set_graphical_culture = pirate_01
		set_country_flag = gk_privateer_enclave_country
		remove_country_flag = mercenary_enclave_country

		# Assign personality
		if = {
			limit = { exists = event_target:gk_privateer_patron }
			event_target:gk_privateer_patron = {
				save_event_target_as = gk_privateer_enclave_creator
			}
		}
		gk_assign_privateer_personality = yes

		# Replace station: destroy merc station, create privateer station at same location
		random_owned_fleet = {
			limit = { is_ship_size = enclave_mercenary_station }
			solar_system = {
				save_event_target_as = gk_station_system
			}
			delete_fleet = this
		}
		create_fleet = {
			settings = { spawn_debris = no }
			effect = {
				set_owner = prev
				create_ship = {
					name = random
					design = "NAME_GK_Privateer_Enclave_Station"
					graphical_culture = prev
				}
				set_location = {
					target = event_target:gk_privateer_enclave_system
					distance = 90
				}
			}
		}

		# End merc dividends, start privateer dividends
		random_situation = {
			limit = { is_situation_type = mercenary_dividends }
			destroy_situation = this
		}
		if = {
			limit = { exists = event_target:gk_privateer_patron }
			start_situation = {
				type = gk_privateer_dividends
				target = event_target:gk_privateer_patron
			}
		}

		# Notify patron
		if = {
			limit = { exists = event_target:gk_privateer_patron }
			event_target:gk_privateer_patron = {
				country_event = { id = gk_raiding_privateers.114 }
			}
		}
	}
}

# Privateer -> Merc: execute conversion — hidden
# This = enclave country (was gk_privateer_enclave, becomes enclave_mercenary)
# Fired from gk_raiding_privateers.205
country_event = {
	id = gk_raiding_privateers.113
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		# Find patron before conversion
		gk_get_privateer_station_context = yes

		# Change country type and civic
		set_country_type = enclave_mercenary
		force_remove_civic = civic_gk_privateer_enclave
		force_add_civic = civic_mercenary_enclave
		if = {
			limit = { exists = event_target:gk_privateer_patron }
			set_graphical_culture = event_target:gk_privateer_patron
		}
		remove_country_flag = gk_privateer_enclave_country
		set_country_flag = mercenary_enclave_country

		# Clear personality flags
		remove_country_flag = gk_privateer_stoic
		remove_country_flag = gk_privateer_screamer
		remove_country_flag = gk_privateer_cultist
		remove_country_flag = gk_privateer_calculating
		remove_country_flag = gk_privateer_merchant
		remove_country_flag = gk_privateer_automaton

		# Replace station: destroy privateer station, create merc station at same location
		random_owned_fleet = {
			limit = {
				any_owned_ship = { is_ship_size = gk_privateer_station }
			}
			delete_fleet = this
		}
		create_fleet = {
			settings = { spawn_debris = no }
			effect = {
				set_owner = prev
				create_ship = {
					name = random
					design = "NAME_Mercenary_Enclave_Station"
					graphical_culture = prev
				}
				set_location = {
					target = event_target:gk_privateer_enclave_system
					distance = 90
				}
			}
		}

		# End privateer dividends, start merc dividends
		random_situation = {
			limit = { is_situation_type = gk_privateer_dividends }
			destroy_situation = this
		}
		if = {
			limit = { exists = event_target:gk_privateer_patron }
			start_situation = {
				type = mercenary_dividends
				target = event_target:gk_privateer_patron
			}
		}

		# Notify patron
		if = {
			limit = { exists = event_target:gk_privateer_patron }
			event_target:gk_privateer_patron = {
				country_event = { id = gk_raiding_privateers.115 }
			}
		}
	}
}

# Station Destruction (600–609)
# ──────────────────────────────────────────────────────────────────

# 600 — Hidden: Privateer station destroyed
# Fired from on_ship_destroyed_victim
# This = enclave country (owner of destroyed ship)
# From = destroyer country
# FromFrom = destroyed ship
country_event = {
	id = gk_raiding_privateers.600
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_country_type = gk_privateer_enclave
		fromfrom = {
			is_ship_size = gk_privateer_station
		}
	}

	immediate = {
		save_event_target_as = gk_privateer_enclave_country

		# Save the system for notification loc
		fromfrom = {
			solar_system = {
				save_event_target_as = gk_privateer_enclave_system
				remove_star_flag = enclave
			}
		}

		# Save destroyer
		from = {
			save_event_target_as = gk_privateer_destroyer
		}

		# Find patron
		gk_get_privateer_station_context = yes

		# If a raid is active, clean up the temp raider
		if = {
			limit = {
				has_country_flag = gk_auto_raid_ongoing
				any_country = {
					is_country_type = gk_privateer_raiders
					has_country_flag = gk_privateer_raider_of_@root
				}
			}
			random_country = {
				limit = {
					is_country_type = gk_privateer_raiders
					has_country_flag = gk_privateer_raider_of_@root
				}
				save_event_target_as = gk_privateer_raider
			}
			gk_cleanup_privateer_raid = yes
		}

		# Remove patron flag
		if = {
			limit = { exists = event_target:gk_privateer_patron }
			event_target:gk_privateer_patron = {
				remove_country_flag = gk_is_privateer_patron
			}
		}

		# Destroy the dividend situation
		every_situation = {
			limit = { is_situation_type = gk_privateer_dividends }
			destroy_situation = this
		}

		# Notify patron
		if = {
			limit = {
				exists = event_target:gk_privateer_patron
				NOT = {
					event_target:gk_privateer_patron = {
						is_same_value = event_target:gk_privateer_destroyer
					}
				}
			}
			event_target:gk_privateer_patron = {
				country_event = { id = gk_raiding_privateers.601 }
			}
		}

		# Notify destroyer
		if = {
			limit = {
				exists = event_target:gk_privateer_destroyer
				event_target:gk_privateer_destroyer = {
					is_country_type = default
				}
			}
			event_target:gk_privateer_destroyer = {
				country_event = { id = gk_raiding_privateers.602 }
			}
		}

		# Notify all communicating empires (except patron and destroyer)
		every_country = {
			limit = {
				has_communications = root
				is_country_type = default
				NOT = { is_same_value = event_target:gk_privateer_destroyer }
				NOT = { is_same_value = event_target:gk_privateer_patron }
			}
			country_event = { id = gk_raiding_privateers.603 }
		}
	}

	after = {
		destroy_country = yes
		# Restore enclave capacity for all empires that have the modifier
		every_country = {
			limit = {
				has_modifier = country_enclave_capacity_decrease_modifier
			}
			country_event = { id = enclave.24 days = 2 }
		}
	}
}

# ──────────────────────────────────────────────────────────────────
# Patron Change / System Ownership Validation (610–619)
# ──────────────────────────────────────────────────────────────────

# 610 — Monthly: check if the station's system ownership has changed
# This = enclave country (from on_monthly_pulse_country)
country_event = {
	id = gk_raiding_privateers.610
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_country_type = gk_privateer_enclave
		# Only fire if we have a station
		any_owned_fleet = {
			any_owned_ship = {
				is_ship_size = gk_privateer_station
			}
		}
	}

	immediate = {
		save_event_target_as = gk_privateer_enclave_country

		# Find station system, star, and current space_owner
		gk_get_privateer_station_context = yes
		# Alias: gk_privateer_patron here is the current system owner, not necessarily the patron
		if = {
			limit = { exists = event_target:gk_privateer_patron }
			event_target:gk_privateer_patron = {
				save_event_target_as = gk_current_system_owner
			}
		}

		# Get old patron from the dividend situation target (authoritative source)
		random_situation = {
			limit = { is_situation_type = gk_privateer_dividends }
			target = {
				save_event_target_as = gk_old_patron
			}
		}

		# Case 1: System has a new owner different from old patron
		if = {
			limit = {
				exists = event_target:gk_current_system_owner
				exists = event_target:gk_old_patron
				NOT = {
					event_target:gk_current_system_owner = {
						is_same_value = event_target:gk_old_patron
					}
				}
			}
			# Remove old patron flag
			event_target:gk_old_patron = {
				remove_country_flag = gk_is_privateer_patron
			}

			# Destroy old dividend situation
			every_situation = {
				limit = { is_situation_type = gk_privateer_dividends }
				destroy_situation = this
			}

			# Set new patron
			event_target:gk_current_system_owner = {
				set_country_flag = gk_is_privateer_patron
			}

			# Start new dividend situation
			start_situation = {
				type = gk_privateer_dividends
				target = event_target:gk_current_system_owner
			}

			# Notify old patron
			event_target:gk_old_patron = {
				country_event = { id = gk_raiding_privateers.611 }
			}

			# Notify new patron
			event_target:gk_current_system_owner = {
				country_event = { id = gk_raiding_privateers.612 }
			}
		}
		# Case 2: System has no owner and we had an old patron
		else_if = {
			limit = {
				NOT = { exists = event_target:gk_current_system_owner }
				exists = event_target:gk_old_patron
			}
			# Remove old patron flag
			event_target:gk_old_patron = {
				remove_country_flag = gk_is_privateer_patron
			}

			# Destroy dividend situation (no patron to pay)
			every_situation = {
				limit = { is_situation_type = gk_privateer_dividends }
				destroy_situation = this
			}

			# Notify old patron
			event_target:gk_old_patron = {
				country_event = { id = gk_raiding_privateers.611 }
			}
		}
		# Case 3: System has an owner but no dividend situation (patron was destroyed)
		else_if = {
			limit = {
				exists = event_target:gk_current_system_owner
				NOT = { exists = event_target:gk_old_patron }
				NOT = {
					any_situation = {
						is_situation_type = gk_privateer_dividends
					}
				}
			}
			event_target:gk_current_system_owner = {
				set_country_flag = gk_is_privateer_patron
			}
			start_situation = {
				type = gk_privateer_dividends
				target = event_target:gk_current_system_owner
			}
			event_target:gk_current_system_owner = {
				country_event = { id = gk_raiding_privateers.612 }
			}
		}
	}
}
