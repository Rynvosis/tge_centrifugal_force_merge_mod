namespace = better_colony_dates

event = {
    id = better_colony_dates.0
    hide_window = yes

    is_triggered_only = yes

    # This is only run at game start
    immediate = {
        every_galaxy_planet = {
            limit = { is_colony = yes }
            planet_event = { id = better_colony_dates.1 }
        }
        # This event catches any planets that were spawned by event during game start, as they'll slip through.
        # Not actually a planet events, works scopelessly. Just done like this to let the simulation tick a day.
        random_galaxy_planet = {
            planet_event = { id = better_colony_dates.6 days = 1 }
        }
    }
}

planet_event = {
    id = better_colony_dates.1
    hide_window = yes

    is_triggered_only = yes

    # This is only run at game start
    immediate = {
       if = {
           limit = { owner = { is_country_type = default } }
            planet_event = { id = better_colony_dates.2 }
       }
        else_if = {
           limit = { owner = { is_country_type = fallen_empire } }
            planet_event = { id = better_colony_dates.3 }
        }
        else_if = {
           limit = { owner = { is_country_type = primitive } }
            planet_event = { id = better_colony_dates.4 }
        }
        # I have absolutely no idea what might be caught here, but it's good to have a catchall.
        else = {
            planet_event = { id = better_colony_dates.5 }
        }
    }
}

# Default empire planet
planet_event = {
    id = better_colony_dates.2
    hide_window = yes

    is_triggered_only = yes

    # This is only run at game start
    immediate = {
        if = {
           limit = { owner = { has_country_flag = is_advanced_start_AI } }
            # Homeworlds should be decently older than a typical tag to justify having colonies
            if = {
                limit = {  is_homeworld = yes }
                bcfd_change_colony_date_range = {
                    MAX_YEARS = -6000
                    MIN_YEARS = -12000
                }
            }
            # Advanced Empire Colonies
            else = {
                # Lost colony shouldn't be that old - UNE shouldn't have colonies from the 1600s
                if = {
                    limit = { owner = { has_country_flag = lost_colony_parent } }
                    bcfd_change_colony_date_range = {
                        MAX_YEARS = -80
                        MIN_YEARS = -150
                    }
                }
                # Normal advanced empire
                else = {
                    bcfd_change_colony_date_range = {
                        MAX_YEARS = -150
                        MIN_YEARS = -600
                    }
                }
            }
        }
        else_if = {
           limit = { owner = { has_origin = origin_lost_colony } }
            bcfd_change_colony_date_range = {
                MAX_YEARS = -75
                MIN_YEARS = -175
            }
        }
        else_if = {
           limit = { owner = { has_origin = origin_shattered_ring } }
            bcfd_change_colony_date_range = {
                MAX_YEARS = -8000
                MIN_YEARS = -12000
            }
        }
        else_if = {
           limit = { owner = { has_origin = origin_remnants } }
            bcfd_change_colony_date_range = {
                MAX_YEARS = -8000
                MIN_YEARS = -12000
            }
        }
        else_if = {
           limit = { owner = { has_origin = origin_hegemon_npc } }
            # A little bit less than the base homeworld option as the overlord would have been more advanced to subjugate.
            bcfd_change_colony_date_range = {
                MAX_YEARS = -5000
                MIN_YEARS = -6000
            }
        }
        else_if = {
           limit = {
               AND = {
                   owner = { has_origin = origin_toxic_knights }
                   is_artificial = yes
               }
           }
            # Toxic god stations should be a bit younger.
            bcfd_change_colony_date_range = {
                MAX_YEARS = -200
                MIN_YEARS = -300
            }
        }
        # PD Domed Hab Start Compatibility
        else_if = {
            limit = { has_planet_flag = pd_habitat_start_colony  }
            bcfd_change_colony_date_range = {
                MAX_YEARS = -40
                MIN_YEARS = -120
            }
        }
        # Catches habs + any other potential artificial starts
        else_if = {
           limit = { is_artificial = yes }
            bcfd_change_colony_date_range = {
                MAX_YEARS = -400
                MIN_YEARS = -2000
            }
        }
        else = {
            # Default option for homeworlds
            if = {
                limit = { is_homeworld = yes }
                bcfd_change_colony_date_range = {
                    MAX_YEARS = -6000
                    MIN_YEARS = -10000
                }
            }
            # Final Catchall
            else = {
                bcfd_change_colony_date_range = {
                    MAX_YEARS = -100
                    MIN_YEARS = -500
                }
            }
        }
    }
}

    id = better_colony_dates.7
    hide_window = yes

    is_triggered_only = yes

    # This is only run at game start
    immediate = {
        # Distant Origin Vestigial Colony Compatibility
        if = {

            bcfd_reset_colony_date = yes
            bcfd_change_colony_date_range = {

                MIN_YEARS = -6000
            }
        }
    }
