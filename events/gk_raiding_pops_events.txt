# Pop abducted by raiding bombardment stance
# This = Planet where pop was "killed" (source)
# From = Bombarder country
# FromFrom = Biggest fleet bombarding
# local_kill_pop_amount = exact number of pops transferred
planet_event = {
	id = gk_raiding_pops.100
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_orbital_bombardment = yes
		has_orbital_bombardment_stance = raiding
	}

	immediate = {
		from = {
			save_event_target_as = gk_raider
		}
		owner = {
			save_event_target_as = gk_raidee
		}
		# Accumulate abduction amount for the redirect event
		from = {
			change_variable = { which = gk_pending_abduction_amount value = root.local_kill_pop_amount }
		}
		# Resolve side leader for bucket tracking
		from = {
			gk_save_pop_raid_side_leader_against_target = yes
		}
		owner = {
			gk_track_raided_pops = { AMOUNT = root.local_kill_pop_amount }
		}
	}
}

# Redirect engine-abducted pops to a preferred destination
# This = Pop Group (already merged onto engine's random pick)
# From = Planet abducted from
pop_group_event = {
	id = gk_raiding_pops.105
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = owner
		owner = {
			is_variable_set = gk_pending_abduction_amount
			check_variable = { which = gk_pending_abduction_amount value > 0 }
		}
	}

	immediate = {
		owner = {
			save_event_target_as = gk_raider
		}
		save_event_target_as = gk_source_pop_group

		# Find preferred destination
		gk_find_destination_planet = yes

		# Skip if the engine already picked our preferred planet
		if = {
			limit = {
				exists = event_target:gk_destination_planet
				planet = {
					NOT = { is_same_value = event_target:gk_destination_planet }
				}
			}
			# Transfer only the abducted amount to the preferred planet
			event_target:gk_destination_planet = {
				create_pop_group = {
					pop_group = event_target:gk_source_pop_group
					size = 0
					effect = {
						transfer_pop_amount = {
							source = event_target:gk_source_pop_group
							target = this
							amount = event_target:gk_raider.gk_pending_abduction_amount
						}
					}
				}
			}
		}

		# Clear pending amount
		owner = {
			clear_variable = gk_pending_abduction_amount
		}
	}
}

# Monthly auto-raid: forcefully resettle pops from occupied enemy planets
# This = Country (from on_monthly_pulse_country)
# Only the side leader processes this to avoid double-counting
country_event = {
	id = gk_raiding_pops.130
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_at_war = yes
		gk_can_pop_raid_personally = yes
	}

	immediate = {
		save_event_target_as = gk_raider
		# For each country we're pop-raiding
		every_country = {
			limit = {
				is_at_war_with = root
				# We have a pop-raiding war against them
				OR = {
					gk_has_standard_raiding_war_with = yes
					gk_has_emancipation_war_with = yes
					gk_has_reaping_war_with = yes
					gk_has_opportunistic_raiding_with = yes
				}
				# Cap not reached and not terminated
				NOT = { has_country_flag = gk_raid_cap_reached_by_@root }
				NOT = { has_country_flag = gk_pop_raiding_terminated_by_@root }
				# No pending offers
				NOT = { has_country_flag = gk_tribute_offer_pending_from_@root }
				NOT = { has_country_flag = gk_reforms_offer_pending_from_@root }
			}
			save_event_target_as = gk_raidee
			root = {
				save_event_target_as = gk_cap_raider
				gk_save_pop_raid_side_leader_against_target = yes
			}
			# Only process if we ARE the side leader (avoid allies double-processing)
			if = {
				limit = {
					event_target:gk_cap_raider = { is_same_value = root }
				}
				# Process each occupied planet
				every_owned_planet = {
					limit = {
						is_controlled_by = root
						any_owned_pop_group = {
							pop_group_size > 0
							is_sapient = yes
							gk_can_abduct_pop_group_war_goal_check = yes
							gk_can_abduct_pop_group_robot_check = yes
						}
					}
					save_event_target_as = gk_target_planet
					save_event_target_as = gk_raiding_target
					add_modifier = { modifier = gk_planet_auto_raid days = 35 }
					root = {
						# Per-army-owner auto-raid with monthly cap
						gk_raid_per_army_owner = { MONTHS = 1 }

						# Notify if pops were taken (throttled to once per 6 months per planet)
						if = {
							limit = {
								check_variable = { which = gk_forfeited_pops value > 0 }
								event_target:gk_target_planet = {
									NOT = { has_planet_flag = gk_recently_auto_raided }
								}
							}
							event_target:gk_target_planet = {
								set_timed_planet_flag = {
									flag = gk_recently_auto_raided
									months = 6
								}
							}
							create_message = {
								type = MESSAGE_TYPE_GK_POP_RAIDED
								localization = MESSAGE_GK_POP_RAIDED_OCCUPATION
								days = 30
								target = event_target:gk_target_planet
								variable = {
									type = name
									localization = PLANET
									scope = event_target:gk_target_planet
								}
								variable = {
									type = variable
									localization = POPS
									varname = gk_forfeited_pops
									scope = root
								}
							}
							event_target:gk_raidee = {
								create_message = {
									type = MESSAGE_TYPE_GK_OUR_POPS_RAIDED
									localization = MESSAGE_GK_OUR_POPS_RAIDED_OCCUPATION
									days = 30
									target = event_target:gk_target_planet
									variable = {
										type = name
										localization = PLANET
										scope = event_target:gk_target_planet
									}
									variable = {
										type = name
										localization = FROM
										scope = root
									}
									variable = {
										type = variable
										localization = POPS
										varname = gk_forfeited_pops
										scope = root
									}
								}
							}
						}

						# Check if planet depopulated
						if = {
							limit = {
								event_target:gk_target_planet = {
									count_owned_pop_amount = { count = 0 }
								}
							}
							# Only show planet cleared if raidee still has pops elsewhere
							if = {
								limit = {
									event_target:gk_raidee = {
										count_owned_pop_amount = { count > 0 }
									}
								}
								country_event = {
									id = gk_raiding_pops.135
									scopes = { from = event_target:gk_target_planet }
								}
								event_target:gk_raidee = {
									country_event = {
										id = gk_raiding_pops.136
										scopes = { from = root }
									}
								}
							}
							# Check empire-wide depopulation
							if = {
								limit = {
									event_target:gk_raidee = {
										count_owned_pop_amount = { count = 0 }
									}
								}
								country_event = {
									id = gk_raiding_pops.215
									scopes = { from = event_target:gk_raidee }
								}
							}
						}
					}
				}
			}
		}
	}
}

# Yearly cleanup of raid tracker leaders and cap flags
# This = Country scope (from on_yearly_pulse_country)
country_event = {
	id = gk_raiding_pops.120
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		any_country = {
			OR = {
				has_country_flag = gk_raid_cap_reached_by_@root
				has_country_flag = gk_pop_raiding_terminated_by_@root
				has_country_flag = gk_opportunistic_raid_authorized_by_@root
				root = { has_country_flag = gk_pivoted_to_pillaging_against_@prev }
			}
		}
	}

	immediate = {
		every_country = {
			limit = {
				OR = {
					has_country_flag = gk_raid_cap_reached_by_@root
					has_country_flag = gk_pop_raiding_terminated_by_@root
					has_country_flag = gk_tribute_offer_pending_from_@root
					has_country_flag = gk_reforms_offer_pending_from_@root
					has_country_flag = gk_autonomy_offer_pending_from_@root
					has_country_flag = gk_opportunistic_raid_authorized_by_@root
					root = { has_country_flag = gk_pivoted_to_pillaging_against_@prev }
				}
			}
			if = {
				limit = {
					NOT = {
						root = {
							any_war = {
								is_war_participant = {
									who = prevprev
								}
							}
						}
					}
				}
				remove_country_flag = gk_raid_cap_reached_by_@root
				remove_country_flag = gk_pop_raiding_terminated_by_@root
				remove_country_flag = gk_tribute_offer_pending_from_@root
				remove_country_flag = gk_reforms_offer_pending_from_@root
				remove_country_flag = gk_autonomy_offer_pending_from_@root
				remove_country_flag = gk_opportunistic_raid_authorized_by_@root
				root = {
					remove_country_flag = gk_pivoted_to_pillaging_against_@prev
					remove_country_flag = gk_opportunistic_prompted_@prev
				}
			}
		}
		save_event_target_as = gk_raider
		event_target:global_event_country = {
			every_owned_leader = {
				limit = {
					has_leader_flag = gk_raid_tracker
					has_leader_flag = gk_raid_tracker_attacker_@root
				}
				if = {
					limit = {
						root = { is_at_war = no }
					}
					# Kill contributor trackers linked to this raid tracker
					save_event_target_as = gk_dying_tracker
					event_target:global_event_country = {
						every_owned_leader = {
							limit = {
								has_leader_flag = gk_contributor_tracker
								has_leader_flag = gk_contributor_for_tracker_@event_target:gk_dying_tracker
							}
							kill_leader = { show_notification = no }
						}
					}
					kill_leader = { show_notification = no }
				}
			}
		}
	}
}

# On planet attackers win - instant pop raid
# This = Attacking Country
# From = Defending Country
# FromFrom = Planet
country_event = {
	id = gk_raiding_pops.210
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		fromfrom = {
			is_colony = yes
		}
		gk_can_pop_raid_personally = yes
		from = {
			OR = {
				gk_has_standard_raiding_war_with = yes
				gk_has_emancipation_war_with = yes
				gk_has_reaping_war_with = yes
				gk_has_opportunistic_raiding_with = yes
			}
		}
	}

	immediate = {
		save_event_target_as = gk_raider
		from = {
			save_event_target_as = gk_raidee
		}
		fromfrom = {
			save_event_target_as = gk_target_planet
			save_event_target_as = gk_raiding_target
		}

		# Resolve side leader for bucket tracking
		gk_save_pop_raid_side_leader_against_target = yes
		if = {
			limit = {
				NOT = { exists = event_target:gk_cap_raider }
			}
			save_event_target_as = gk_cap_raider
		}

		# Per-army-owner invasion burst (accumulates gk_forfeited_pops on root)
		gk_invasion_per_army_owner = { MONTHS = @gk_invasion_months }

		if = {
			limit = {
				check_variable = { which = gk_forfeited_pops value > 0 }
			}
			# Notify attacker
			create_message = {
				type = MESSAGE_TYPE_GK_POP_RAIDED
				localization = MESSAGE_GK_POP_RAIDED_INVASION
				days = 30
				target = event_target:gk_target_planet
				variable = {
					type = name
					localization = PLANET
					scope = event_target:gk_target_planet
				}
				variable = {
					type = variable
					localization = POPS
					varname = gk_forfeited_pops
					scope = root
				}
			}
			# Notify defender
			event_target:gk_raidee = {
				create_message = {
					type = MESSAGE_TYPE_GK_OUR_POPS_RAIDED
					localization = MESSAGE_GK_OUR_POPS_RAIDED_INVASION
					days = 30
					target = event_target:gk_target_planet
					variable = {
						type = name
						localization = PLANET
						scope = event_target:gk_target_planet
					}
					variable = {
						type = name
						localization = FROM
						scope = root
					}
					variable = {
						type = variable
						localization = POPS
						varname = gk_forfeited_pops
						scope = root
					}
				}
			}
		}

		# Notify war leader when an ally captures pops on their behalf
		if = {
			limit = {
				check_variable = { which = gk_forfeited_pops value > 0 }
				exists = event_target:gk_cap_raider
				event_target:gk_cap_raider = {
					NOT = { is_same_value = root }
				}
			}
			event_target:gk_cap_raider = {
				create_message = {
					type = MESSAGE_TYPE_GK_POP_RAIDED
					localization = MESSAGE_GK_POP_RAIDED_ALLY_INVASION
					days = 30
					target = event_target:gk_target_planet
					variable = {
						type = name
						localization = ALLY
						scope = root
					}
					variable = {
						type = name
						localization = PLANET
						scope = event_target:gk_target_planet
					}
					variable = {
						type = variable
						localization = POPS
						varname = gk_forfeited_pops
						scope = root
					}
				}
			}
		}

		# Check if we just depopulated the raidee entirely
		if = {
			limit = {
				event_target:gk_raidee = {
					count_owned_pop_amount = { count = 0 }
				}
			}
			country_event = {
				id = gk_raiding_pops.215
				scopes = { from = event_target:gk_raidee }
			}
		}

	}
}

						is_war_participant = { who = prev side = defenders }

				NOR = {
					is_total_war = yes

			# Set prompted flag immediately so subsequent bombardment ticks
			# short-circuit before reaching the expensive any_war evaluation

				factor = 2

		}

		add_resource = {
			influence = 1
			mult = value:gk_escalation_influence_cost
		}
		hidden_effect = {
			event_target:gk_raid_tracker = {
				clear_variable = gk_dep_energy
				clear_variable = gk_dep_food
				clear_variable = gk_dep_alloys
				clear_variable = gk_tribute_base
			}
			# Switch war goal to humiliation
			every_war = {
				limit = {
					OR = {
						AND = {
							is_war_participant = { who = root side = attackers }
							is_war_participant = { who = from side = defenders }
							using_war_goal = { type = wg_plunder owner = attacker }
						}
						AND = {
							is_war_participant = { who = root side = defenders }
							is_war_participant = { who = from side = attackers }
							using_war_goal = { type = wg_plunder owner = defender }
						}
					}
				}
				root = {
					set_war_goal = { type = wg_humiliation target = from enemy = from }
				}
			}
		}
	}
