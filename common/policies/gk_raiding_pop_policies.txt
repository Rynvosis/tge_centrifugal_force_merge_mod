# GK Raiding Pop Policies
# Determines pop-raiding focus and robot capture rules

gk_raiding_focus = {
	potential = {
		OR = {
			has_valid_civic = civic_barbaric_despoilers
			has_ascension_perk = ap_nihilistic_acquisition
			has_valid_civic = civic_gk_emancipators
			has_valid_civic = civic_gk_emancipators_megacorp
			has_valid_civic = civic_gk_barbaric_despoilers_machine
			has_valid_civic = civic_gk_barbaric_despoilers_megacorp
			has_valid_civic = civic_hive_bodysnatcher
		}
	}

	allow = {
		is_at_war = no
	}

	option = {
		name = "gk_raiding_focus_plunder"

		policy_flags = {
			gk_focus_plunder
		}

		AI_weight = {
			factor = 2
		}
	}

	option = {
		name = "gk_raiding_focus_emancipation"

		valid = {
			OR = {
				has_policy_flag = slavery_not_allowed
				has_valid_civic = civic_machine_servitor
			}
		}

		policy_flags = {
			gk_focus_emancipation
		}

		AI_weight = {
			factor = 2
			modifier = {
				factor = 3
				has_ethic = ethic_fanatic_egalitarian
			}
			modifier = {
				factor = 2
				has_valid_civic = civic_gk_emancipators
			}
			modifier = {
				factor = 3
				has_valid_civic = civic_machine_servitor
			}
		}
	}

	option = {
		name = "gk_raiding_focus_reaping"

		valid = {
			gk_can_use_reaping_cb = yes
		}

		policy_flags = {
			gk_focus_reaping
		}

		AI_weight = {
			factor = 5
			modifier = {
				factor = 2
				has_ethic = ethic_fanatic_xenophobe
			}
		}
	}
}

# Policy to determine if robots can be raided
gk_robot_raiding = {
	potential = {
		OR = {
			has_valid_civic = civic_barbaric_despoilers
			has_ascension_perk = ap_nihilistic_acquisition
			has_valid_civic = civic_gk_emancipators
			has_valid_civic = civic_gk_emancipators_megacorp
			has_valid_civic = civic_gk_barbaric_despoilers_machine
			has_valid_civic = civic_gk_barbaric_despoilers_megacorp
			has_valid_civic = civic_hive_bodysnatcher
		}
	}

	allow = {
		is_at_war = no
	}

	option = {
		name = "gk_robot_raiding_allowed"

		policy_flags = {
			gk_allow_robot_raiding
		}

		AI_weight = {
			factor = 2
			modifier = {
				factor = 3
				has_ethic = ethic_materialist
			}
			modifier = {
				factor = 3
				has_ethic = ethic_fanatic_materialist
			}
		}
	}

	option = {
		name = "gk_robot_raiding_forbidden"

		policy_flags = {
			gk_forbid_robot_raiding
		}

		AI_weight = {
			factor = 1
			modifier = {
				factor = 3
				has_ethic = ethic_spiritualist
			}
			modifier = {
				factor = 3
				has_ethic = ethic_fanatic_spiritualist
			}
		}
	}

	option = {
		name = "gk_robot_raiding_only"

		valid = {
			NOT = { has_policy_flag = gk_focus_emancipation }
			OR = {
				has_country_flag = synthetic_empire
				has_ethic = ethic_materialist
				has_ethic = ethic_fanatic_materialist
				has_ascension_perk = ap_the_flesh_is_weak
				has_ascension_perk = ap_organo_machine_interfacing
				has_origin = origin_cybernetic_creed
				is_machine_empire = yes
			}
		}

		policy_flags = {
			gk_robots_only_raiding
		}

		AI_weight = {
			factor = 1
			modifier = {
				factor = 5
				has_ethic = ethic_fanatic_materialist
			}
		}
	}
}
