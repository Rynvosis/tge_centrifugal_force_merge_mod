# GK Raiding Pop Scripted Triggers

#############################
# Unlimited Raiding Check   #
#############################

# Checks if raiding should bypass all bucket caps
# This = raidee country
# event_target:gk_cap_raider = side leader
# event_target:gk_raid_tracker = tracker (optional, checked for tier 5)
gk_has_unlimited_raiding = {
	OR = {
		# Tier 5: total depopulation authorized
		AND = {
			exists = event_target:gk_raid_tracker
			event_target:gk_raid_tracker = {
				check_variable = { which = gk_escalation_tier value >= 5 }
			}
		}
		# Reaping war goal
		event_target:gk_cap_raider = {
			any_war = {
				OR = {
					AND = {
						is_war_participant = { who = prev side = attackers }
						is_war_participant = { who = prevprev side = defenders }
						using_war_goal = { type = wg_gk_reaping owner = attacker }
					}
					AND = {
						is_war_participant = { who = prev side = defenders }
						is_war_participant = { who = prevprev side = attackers }
						using_war_goal = { type = wg_gk_reaping owner = defender }
					}
				}
			}
		}
	}
}

# Raiding Bombardment Check #
#############################
gk_can_use_raiding_bombardment = {
	owner = {
		OR = {
			has_valid_civic = civic_gk_emancipators
			has_valid_civic = civic_gk_emancipators_megacorp
			has_valid_civic = civic_gk_barbaric_despoilers_machine
			has_valid_civic = civic_gk_barbaric_despoilers_megacorp
			has_valid_civic = civic_hive_bodysnatcher
		}
	}
}



####################
# CB Access Checks #
####################
# This = country scope
# Can use standard raiding CB (from civics, ascension perks, etc.)
gk_can_use_raiding_cb = {
	OR = {
		has_valid_civic = civic_barbaric_despoilers
		has_ascension_perk = ap_nihilistic_acquisition
		has_valid_civic = civic_gk_emancipators
		has_valid_civic = civic_gk_emancipators_megacorp
		has_valid_civic = civic_gk_barbaric_despoilers_machine
		has_valid_civic = civic_gk_barbaric_despoilers_megacorp
		has_valid_civic = civic_hive_bodysnatcher
	}
}

# This = country scope
# Personal pop-raiding capability.
gk_can_pop_raid_personally = {
	gk_can_use_raiding_cb = yes
}

# This = country scope
# Can use Emancipation war goal (requires emancipation policy)
gk_can_use_emancipation_cb = {
	has_policy_flag = gk_focus_emancipation
}

# This = country scope
# Can use Reaping war goal
gk_can_use_reaping_cb = {
	OR = {
		# Despoilers require xenophobe
		AND = {
			OR = {
				has_valid_civic = civic_barbaric_despoilers
				has_valid_civic = civic_gk_barbaric_despoilers_megacorp
			}
			is_xenophobe = yes
		}
		# Nihilistic Acquisition requires xenophobe
		AND = {
			has_ascension_perk = ap_nihilistic_acquisition
			is_xenophobe = yes
		}
		# Gestalt despoilers always qualify
		has_valid_civic = civic_gk_barbaric_despoilers_machine
		has_valid_civic = civic_hive_bodysnatcher
	}
}

# War-type specific "war with" triggers
# This = country scope (defender/raidee)
# Root = country scope (attacker/raider)
gk_has_standard_raiding_war_with = {
	root = {
		any_war = {
			OR = {
				AND = {
					is_war_participant = { who = prev side = attackers }
					is_war_participant = { who = prevprev side = defenders }
					using_war_goal = { type = wg_plunder owner = attacker }
				}
				AND = {
					is_war_participant = { who = prev side = defenders }
					is_war_participant = { who = prevprev side = attackers }
					using_war_goal = { type = wg_plunder owner = defender }
				}
			}
		}
	}
}

# This = raidee country, Root = raider country
gk_has_opportunistic_raiding_with = {

# Pop Group Checks #
####################
# Root = hostile country (attacker/raider)
# This = pop_group
gk_can_abduct_pop_group_war_goal_check = {
	# Only allow pop abduction in raiding contexts:
	# explicit raiding wars, total war, or authorized opportunistic raiding.
	OR = {
		# No formal war with owner (primitives, first contact, etc.) - allow
		planet = {
			owner = {
				is_primitive = yes
			}
		}
		# Total war - always allow
		planet = {
			owner = {
				root = {
					any_war = {
						OR = {
							AND = {
								is_war_participant = {
									who = prev
									side = attackers
								}
								is_war_participant = {
									who = prevprev
									side = defenders
								}
							}
							AND = {
								is_war_participant = {
									who = prev
									side = defenders
								}
								is_war_participant = {
									who = prevprev
									side = attackers
								}
							}
						}
						is_total_war = yes
					}
				}
			}
		}
		# Standard raiding or reaping war with this defender - any pop
		planet = {
			owner = {
				OR = {
					gk_has_standard_raiding_war_with = yes
					gk_has_reaping_war_with = yes
				}
			}
		}
		# Emancipation war with this defender - enslaved only
		AND = {
			planet = { owner = { gk_has_emancipation_war_with = yes } }
			is_enslaved = yes
		}
		# Opportunistic raiding - any pop (plunder/reaping focus)
		AND = {
			planet = { owner = { gk_has_opportunistic_raiding_with = yes } }
			root = { NOT = { has_policy_flag = gk_focus_emancipation } }
		}
		# Opportunistic raiding - enslaved only (emancipation focus)
		AND = {
			planet = { owner = { gk_has_opportunistic_raiding_with = yes } }
			root = { has_policy_flag = gk_focus_emancipation }
			is_enslaved = yes
		}
	}
}

# Root = hostile country (attacker/raider)
# This = pop_group
# Flag-based cap check - all rate logic is in gk_update_raid_cap_flag
gk_can_abduct_pop_group_cap_check = {
	# Terminal state: pop raiding terminated for this raider
	NOT = {
		planet = {
			owner = {
				has_country_flag = gk_pop_raiding_terminated_by_@root
			}
		}
	}
	OR = {
		# Total war - no cap
		planet = {
			owner = {
				root = {
					any_war = {
						OR = {
							AND = {
								is_war_participant = {
									who = prev
									side = attackers
								}
								is_war_participant = {
									who = prevprev
									side = defenders
								}
							}
							AND = {
								is_war_participant = {
									who = prev
									side = defenders
								}
								is_war_participant = {
									who = prevprev
									side = attackers
								}
							}
						}
						is_total_war = yes
					}
				}
			}
		}
		# Primitives - no cap
		planet = {
			owner = {
				is_primitive = yes
			}
		}
		# Reaping war - no cap
		planet = {
			owner = {
				gk_has_reaping_war_with = yes
			}
		}
		# Cap not reached (flag-based check)
		planet = {
			owner = {
				root = {
					any_war = {
						OR = {
							AND = {
								is_war_participant = { who = prev side = attackers }
								is_war_participant = { who = prevprev side = defenders }
								OR = {
									using_war_goal = { type = wg_plunder owner = attacker }
									using_war_goal = { type = wg_gk_emancipation owner = attacker }
								}
								attacker = {
									prevprevprev = {
										NOT = { has_country_flag = gk_raid_cap_reached_by_@prev }
									}
								}
							}
							AND = {
								is_war_participant = { who = prev side = defenders }
								is_war_participant = { who = prevprev side = attackers }
								OR = {
									using_war_goal = { type = wg_plunder owner = defender }
									using_war_goal = { type = wg_gk_emancipation owner = defender }
								}
								defender = {
									prevprevprev = {
										NOT = { has_country_flag = gk_raid_cap_reached_by_@prev }
									}
								}
							}
						}
					}
				}
			}
		}
		# Opportunistic raiding - cap not reached
		planet = {
			owner = {
				gk_has_opportunistic_raiding_with = yes
				NOT = { has_country_flag = gk_raid_cap_reached_by_@root }
			}
		}
	}
}

# Root = hostile country (attacker/raider)
# This = pop_group
gk_can_abduct_pop_group_robot_check = {
	# Check: Robot raiding policy restrictions
	OR = {
		# Robots only policy - ONLY robots
		AND = {
			root = {
				has_policy_flag = gk_robots_only_raiding
			}
			is_robotic_species = yes
		}
		# Forbid robot raiding - ONLY organics
		AND = {
			root = {
				has_policy_flag = gk_forbid_robot_raiding
			}
			is_robotic_species = no
		}
		# Allow robot raiding - both allowed
		AND = {
			root = {
				has_policy_flag = gk_allow_robot_raiding
			}
		}
	}
}

# Root = hostile country (attacker/raider)
# This = pop_group
gk_can_abduct_pop_group = {
	# Vanilla checks the pop is eligible to be raided in the first place.
	# This filters based on the raiding quotas and war goal restrictions.
	# Check 0: No Pop-Raiding policy blocks all pop raiding
	root = {
		gk_can_pop_raid_personally = yes
	}
	# Check 1: War Goal Restrictions
	gk_can_abduct_pop_group_war_goal_check = yes
	# Check 2: Per-Raidee Limit
	gk_can_abduct_pop_group_cap_check = yes
	# Check 3: Robot Raiding Policy
	gk_can_abduct_pop_group_robot_check = yes
}

# This = planet scope
# Check if planet is good destination for raided pops
gk_is_valid_pop_destination = {
	is_colony = yes
	exists = owner
	free_housing >= 1
	free_amenities >= 1
}
