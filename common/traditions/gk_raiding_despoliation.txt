# GK Raiding Despoliation Traditions

tr_gk_raiding_despoliation_adopt = {
	unlocks_agenda = agenda_gk_raiding_despoliation_recovery_fleets
	custom_tooltip = GK_RAIDING_TR_GK_RAIDING_DESPOLIATION_ADOPT_EFFECT

	tradition_swap = {
		name = tr_gk_raiding_despoliation_adopt_native
		inherit_name = yes
		inherit_icon = yes
		trigger = {
			gk_has_native_pillage_access = yes
		}
		modifier = {
			gk_pillage_yield = 1
		}
		custom_tooltip = GK_RAIDING_TR_GK_RAIDING_DESPOLIATION_ADOPT_NATIVE_EFFECT

		weight = {
			factor = 1
		}
	}
}

tr_gk_raiding_despoliation_finish = {
	# Default: no Overlord DLC — combat bonuses instead of enclave capacity
	modifier = {
		ship_fire_rate_mult = 0.05
		army_damage_mult = 0.10
		ascension_perks_add = 1
	}

	# Overlord DLC: privateer enclave access
	tradition_swap = {
		name = tr_gk_raiding_despoliation_finish_overlord
		inherit_name = yes
		inherit_icon = yes
		trigger = {
			has_overlord_dlc = yes
			NOT = { has_origin = origin_mindwardens }
			NOT = { has_civic = civic_fanatic_purifiers }
		}
		modifier = {
			country_enclave_capacity_add = 1
			ascension_perks_add = 1
		}
		custom_tooltip_with_modifiers = GK_RAIDING_TR_GK_RAIDING_DESPOLIATION_FINISH_EFFECT
		weight = { factor = 1 }
	}

	# Overlord DLC + Mindwarden: extra enclave capacity
	tradition_swap = {
		name = tr_gk_raiding_despoliation_finish_mindwarden
		inherit_name = yes
		inherit_icon = yes
		trigger = {
			has_overlord_dlc = yes
			has_origin = origin_mindwardens
		}
		modifier = {
			country_enclave_capacity_add = 2
			ascension_perks_add = 1
		}
		weight = { factor = 2 }
	}

	# Fanatic Purifiers: combat bonuses (same with or without Overlord)
	tradition_swap = {
		name = tr_gk_raiding_despoliation_finish_purifier
		inherit_name = yes
		inherit_icon = yes
		trigger = { has_civic = civic_fanatic_purifiers }
		modifier = {
			ship_fire_rate_mult = 0.05
			army_damage_mult = 0.10
			ascension_perks_add = 1
		}
		weight = { factor = 3 }
	}
}
