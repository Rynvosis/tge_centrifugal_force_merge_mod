ethic_individualist = {
	cost = 1
	category = "ind"
	category_value = 1
	fanatic_variant = ethic_fanatic_individualist
	leader_background_job_weight = {
		merchant = 25
		clerk = 25
		artisan = 25
		artificer = 25
		executive = 25
		entertainer = 25
	}
	tags = {
		ETHIC_ALLOWS_STRATIFIED_SOCIETY
	}
	country_modifier = {
		pop_cat_ruler_bonus_workforce_mult = 0.10
		country_trade_fee = -0.05
		add_attunement_the_instrument_of_desire = @ethic_attunement1
	}
	random_weight = {
		base = 100
	}
	pop_attraction_tag = {
		desc = POP_ATTRACTION_INDIVIDUALIST_ETHIC_POS
		trigger = {
			is_individualist = yes
		}
	}
	pop_attraction_tag = {
		desc = POP_ATTRACTION_FACTION_POS
		trigger = {
			is_individualist = no
			has_faction = gk_ec_individualist
		}
	}
	pop_attraction_tag = {
		desc = POP_ATTRACTION_THRIFTY_POS
		trigger = {
			any_owned_species = {
				has_trait = "trait_thrifty"
			}
		}
	}

	pop_attraction_tag = {
		desc = POP_ATTRACTION_SOLITARY_POS
		trigger = {
			any_owned_species = {
				has_trait = "trait_solitary"
			}
		}
	}

	pop_attraction_tag = {
		desc = POP_ATTRACTION_GOVERNOR_POS
		trigger = {
			any_owned_planet = {
				exists = leader
				leader = { has_ethic = ethic_individualist }
			}
		}
	}

	pop_attraction_tag = {
		desc = POP_ATTRACTION_STATELY_ACCLIAM
		trigger = {
			has_edict = stately_acclaim
			any_pop_faction = {
				has_ethic = ethic_individualist
				is_faction_extorted = yes
			}
		}
	}

	pop_attraction_tag = {
		desc = POP_ATTRACTION_OVERLORD_POS
		trigger = {
			is_subject = yes
			has_loyalty > 0
			overlord = { is_individualist = yes }
		}
	}

	pop_attraction_tag = {
		desc = POP_ATTRACTION_INDIVIDUALIST_DIPLOMACY_POS
		trigger = {
			any_relation = {
				OR = {
					is_in_federation_with = root
					has_defensive_pact = root
					has_commercial_pact = root
				}
				is_individualist = yes
			}
		}
	}

	pop_attraction_tag = {
		desc = POP_ATTRACTION_TRADEFED_POS
		trigger = {
			has_federation = yes
			federation = {
				has_federation_type = trade_federation
			}
		}
	}

	country_attraction = {
		base = 1
		modifier = {
			factor = 0
			NOR = {
				is_country_type = default
				is_individualist = yes
				AND = {
					is_country_type = primitive
					has_country_flag = preftl_societal_enlightenment_pact
					capital_scope = {
						has_observation_outpost = yes
						observation_outpost_owner = { is_individualist = yes }
					}
				}
			}
		}
		# Effects of indoctrination mission
		modifier = {
			factor = 10
			is_country_type = primitive
			has_country_flag = preftl_societal_enlightenment_pact
			capital_scope = {
				has_observation_outpost = yes
				observation_outpost_owner = { is_fanatic_individualist = yes }
			}
		}
		modifier = {
			factor = 5
			is_country_type = primitive
			has_country_flag = preftl_societal_enlightenment_pact
			capital_scope = {
				has_observation_outpost = yes
				observation_outpost_owner = { has_ethic = ethic_individualist }
			}
		}
		modifier = {
			factor = 2
			is_country_type = default
			is_individualist = no
			has_faction = gk_ec_individualist
		}
		modifier = {
			factor = 2
			has_ethic = ethic_individualist
		}
		modifier = {
			factor = 3
			is_fanatic_individualist = yes
		}

	}
	# from = planet
	pop_attraction = {
		base = 1

		inline_script = {
			script = "ethics/governor_pop_attraction"
			ETHIC = individualist
			ETHIC_OPPOSED = collectivist
		}

		inline_script = {
			script = "ethics/stately_acclaim_pop_attraction"
			ETHIC = individualist
		}

		modifier = {
			factor = 0
			has_trait = trait_hive_mind
		}
		modifier = {
			factor = 1.25
			has_trait = "trait_thrifty"
		}
		modifier = {
			factor = 1.1
			has_trait = "trait_solitary"
		}
		modifier = {
			factor = 0.75
			has_trait = "trait_communal"
		}
	}
}
