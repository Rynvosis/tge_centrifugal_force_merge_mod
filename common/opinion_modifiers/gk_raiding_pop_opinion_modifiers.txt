# GK Raiding Pop Opinion Modifiers

# Declared a Reaping war — flat opinion hit on declaration
opinion_gk_declared_reaping = {
	opinion = {
		base = -100
		modifier = {
			set = 0
			OR = {
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire
				is_homicidal = yes
			}
		}
		modifier = {
			add = -50
			has_ethic = ethic_egalitarian
			allows_purge = no
			NOR = {
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire
			}
		}
		modifier = {
			add = -100
			has_ethic = ethic_fanatic_egalitarian
			allows_purge = no
			NOR = {
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire
			}
		}
		modifier = {
			add = -50
			values_others = yes
			allows_purge = no
			NOR = {
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire
			}
		}
		modifier = {
			add = -100
			has_ethic = ethic_fanatic_xenophile
			allows_purge = no
			NOR = {
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire
			}
		}
	}

	decay = {
		base = 0.5
	}
}
