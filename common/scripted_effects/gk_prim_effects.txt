# GK Primitive Start — B-Track Escalation Effects + Year 10 Tech Pools
# This = country scope

gk_prim_advance_exploration = {
	if = {
		limit = { has_country_flag = gk_prim_exploration_2 }
		# 3rd pick: tech at 50%
		set_country_flag = gk_prim_exploration_3
		if = {
			limit = { NOT = { has_technology = tech_sensors_2 } }
			add_tech_progress = {
				tech = tech_sensors_2
				progress = 0.50
			}
		}
	}
	else_if = {
		limit = { has_country_flag = gk_prim_exploration_1 }
		# 2nd pick: upgrade to tier 2 modifier (replaces tier 1)
		set_country_flag = gk_prim_exploration_2
		hidden_effect = { remove_modifier = gk_prim_exploration_1 }
		add_modifier = { modifier = gk_prim_exploration_2 }
	}
	else = {
		# 1st pick: tier 1 modifier
		set_country_flag = gk_prim_exploration_1
		add_modifier = { modifier = gk_prim_exploration_1 }
	}
}

# Year 10 Tech Pool Effects
# Each gives 25% progress in the first 2 unresearched techs from an ordered list.
# Two passes: first sets a temp flag, second skips it.

gk_prim_y10_exploration_tech = {
	# IMPORTANT: Pass 1 and Pass 2 tech lists must stay in sync
	# Pass 1: first unresearched
	if = {
		limit = { NOT = { has_technology = tech_automated_exploration } can_research_technology = tech_automated_exploration }
		give_tech_option_or_progress_effect = { TECH = tech_automated_exploration }
		set_country_flag = gk_prim_y10_gave_first
	}
	else_if = {
		limit = { NOT = { has_technology = tech_administrative_ai } can_research_technology = tech_administrative_ai }
		give_tech_option_or_progress_effect = { TECH = tech_administrative_ai }
		set_country_flag = gk_prim_y10_gave_first
	}
	else_if = {
		limit = { NOT = { has_technology = tech_space_science_1 } can_research_technology = tech_space_science_1 }
		give_tech_option_or_progress_effect = { TECH = tech_space_science_1 }
		set_country_flag = gk_prim_y10_gave_first
	}
	else_if = {
		limit = { NOT = { has_technology = tech_space_science_2 } can_research_technology = tech_space_science_2 }
		give_tech_option_or_progress_effect = { TECH = tech_space_science_2 }
		set_country_flag = gk_prim_y10_gave_first
	}
	else_if = {
		limit = { NOT = { has_technology = tech_basic_science_lab_2 } can_research_technology = tech_basic_science_lab_2 }
		give_tech_option_or_progress_effect = { TECH = tech_basic_science_lab_2 }
		set_country_flag = gk_prim_y10_gave_first
	}
	# Pass 2: second unresearched (skip what pass 1 gave via has_tech_option)
	if = {
		limit = { has_country_flag = gk_prim_y10_gave_first NOT = { has_technology = tech_automated_exploration } can_research_technology = tech_automated_exploration NOT = { has_tech_option = tech_automated_exploration } }
		give_tech_option_or_progress_effect = { TECH = tech_automated_exploration }
	}
	else_if = {
		limit = { has_country_flag = gk_prim_y10_gave_first NOT = { has_technology = tech_administrative_ai } can_research_technology = tech_administrative_ai NOT = { has_tech_option = tech_administrative_ai } }
		give_tech_option_or_progress_effect = { TECH = tech_administrative_ai }
	}
	else_if = {
		limit = { has_country_flag = gk_prim_y10_gave_first NOT = { has_technology = tech_space_science_1 } can_research_technology = tech_space_science_1 NOT = { has_tech_option = tech_space_science_1 } }
		give_tech_option_or_progress_effect = { TECH = tech_space_science_1 }
	}
	else_if = {
		limit = { has_country_flag = gk_prim_y10_gave_first NOT = { has_technology = tech_space_science_2 } can_research_technology = tech_space_science_2 NOT = { has_tech_option = tech_space_science_2 } }
		give_tech_option_or_progress_effect = { TECH = tech_space_science_2 }
	}
	else_if = {
		limit = { has_country_flag = gk_prim_y10_gave_first NOT = { has_technology = tech_basic_science_lab_2 } can_research_technology = tech_basic_science_lab_2 NOT = { has_tech_option = tech_basic_science_lab_2 } }
		give_tech_option_or_progress_effect = { TECH = tech_basic_science_lab_2 }
	}
	remove_country_flag = gk_prim_y10_gave_first
}
