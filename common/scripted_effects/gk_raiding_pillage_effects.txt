# GK Raiding Pillage Effects
# Dedicated file for pillaging mechanics

######################
# Core Pillage Flow  #
######################
# Resource pillaging is now driven by visible devastation:
#   1. Orbital bombardment and ground combat add local_devastation to a per raider+planet receipt bank.
#   2. Every 10 devastation pays out plunder based on additive loot-month sources and flat spoils.
#   3. Invasion and surrender cash out the remaining devastation headroom at the current planet rate
#      without applying additional devastation.
#   4. Surrender only cashes out the first 50 devastation worth on each surviving colony and applies
#      a Ransacked modifier based on current devastation level.

######################
# Entry Points       #
######################

# Pillage for war surrender - remaining surrender-band headroom on all planets
# This = Country (Raider, winner)
# event_target:gk_raidee = Country (Loser)
gk_pillage_on_surrender = {
	save_event_target_as = gk_raider
	event_target:gk_raidee = {
		save_event_target_as = gk_raiding_target
	}

	# Create the country receipt used for surrender display/collection
	gk_get_or_create_loot_receipt = yes
	event_target:gk_loot_receipt = {
		save_event_target_as = gk_storage_scope
	}
	gk_init_loot_variables = yes

	event_target:gk_raidee = {
		every_owned_planet = {
			limit = {
				is_colony = yes
				NOT = { gk_planet_is_ransacked = yes }
			}
			gk_add_surrender_planet_pillage_to_storage = yes
		}
	}

	# Apply conversions and efficiency to the accumulated raw loot
	gk_transform_loot = yes
}

# Pillage for invasion - remaining devastation headroom on a single planet
# This = Country (Raider)
# event_target:gk_target_planet = Planet being invaded
gk_pillage_on_invasion = {
	save_event_target_as = gk_raider
	event_target:gk_target_planet = {
		save_event_target_as = gk_raiding_target
	}

	if = {
		limit = {
			event_target:gk_target_planet = {
				is_colony = yes
				exists = owner
			}
			OR = {
				AND = {
					exists = event_target:gk_raidee
					event_target:gk_raidee = {
						event_target:gk_raider = {
							gk_can_pillage = yes
							is_at_war_with = prev
						}
					}
				}
				AND = {
					NOT = { exists = event_target:gk_raidee }
					event_target:gk_target_planet = {
						owner = {
							event_target:gk_raider = {
								gk_can_pillage = yes
								is_at_war_with = prev
							}
						}
					}
				}
			}
		}

		# Reuse the planet receipt so any unspent partial bank is included
		gk_get_or_create_loot_receipt = yes
		event_target:gk_loot_receipt = {
			save_event_target_as = gk_storage_scope
		}
		gk_init_loot_variables = yes

		if = {
			limit = {
				NOT = { exists = event_target:gk_raidee }
			}
			event_target:gk_target_planet = {
				owner = {
					save_event_target_as = gk_raidee
				}
			}
		}
		event_target:gk_target_planet = {
			set_variable = { which = gk_remaining_pillage_damage_budget value = value:gk_remaining_pillage_budget|MAX|@gk_max_planet_devastation| }
			gk_apply_planet_ransacked_invasion = yes
			if = {
				limit = {
					check_variable = { which = gk_remaining_pillage_damage_budget value > 0 }
				}
				set_variable = {
					which = gk_total_loot_packets
					value = gk_remaining_pillage_damage_budget
				}
				divide_variable = {
					which = gk_total_loot_packets
					value = @gk_pillage_devastation_per_packet
				}
				gk_calculate_base_loot = yes
			}
			clear_variable = gk_remaining_pillage_damage_budget
		}

		# Transform once onto the invasion receipt; collection happens in the event option
		gk_transform_loot = yes
	}
}

# Pillage from orbital bombardment or ground combat devastation - immediate payout
# This = Planet
# event_target:gk_raider = Raider country
gk_pillage_on_devastation = {
	save_event_target_as = gk_raiding_target

	# Use a per planet receipt purely for ownership and the partial devastation bank
	gk_get_or_create_loot_receipt = yes

	# Track fresh devastation on the receipt
	event_target:gk_loot_receipt = {
		change_variable = {
			which = gk_pillage_damage_bank
			value = root.local_devastation
		}
	}

	# Only resolve packets when we have crossed a full 10 devastation threshold
	if = {
		limit = {
			event_target:gk_loot_receipt = {
				check_variable = {
					which = gk_pillage_damage_bank
					value >= @gk_pillage_devastation_per_packet
				}
			}
		}
		set_variable = {
			which = gk_pillage_packet_count
			value = event_target:gk_loot_receipt.gk_pillage_damage_bank
		}
		divide_variable = {
			which = gk_pillage_packet_count
			value = @gk_pillage_devastation_per_packet
		}
		floor_variable = gk_pillage_packet_count

		set_variable = {
			which = gk_pillage_damage_to_consume
			value = gk_pillage_packet_count
		}
		multiply_variable = {
			which = gk_pillage_damage_to_consume
			value = @gk_pillage_devastation_per_packet
		}
		event_target:gk_loot_receipt = {
			subtract_variable = {
				which = gk_pillage_damage_bank
				value = prev.gk_pillage_damage_to_consume
			}
		}

		# Use the planet as temporary storage for the immediate payout + message
		save_event_target_as = gk_storage_scope
		gk_init_loot_variables = yes
		set_variable = {
			which = gk_total_loot_packets
			value = gk_pillage_packet_count
		}
		gk_calculate_base_loot = yes
		gk_transform_loot = yes
		gk_collect_loot_from_storage = yes
		gk_round_loot_variables_for_display = yes
		gk_send_pillage_notifications = yes
		gk_clear_loot_variables = yes

		clear_variable = gk_pillage_packet_count
		clear_variable = gk_pillage_damage_to_consume
	}
}

# Add one planet's remaining surrender payout to the country storage receipt
# This = Planet
# event_target:gk_raider = Winner country
# event_target:gk_raidee = Loser country
# event_target:gk_storage_scope = country receipt
gk_add_surrender_planet_pillage_to_storage = {
	save_event_target_as = gk_target_planet

	set_variable = { which = gk_remaining_pillage_damage_budget value = value:gk_remaining_pillage_budget|MAX|@gk_surrender_planet_devastation| }

	# Fold any saved partial bank into the same remaining surrender-band calculation, then delete the planet receipt
	if = {
		limit = {
			event_target:global_event_country = {
				any_owned_leader = {
					has_leader_flag = gk_loot_receipt_to_@event_target:gk_raider
					has_leader_flag = gk_loot_receipt_from_@event_target:gk_target_planet
				}
			}
		}
		event_target:global_event_country = {
			random_owned_leader = {
				limit = {
					has_leader_flag = gk_loot_receipt_to_@event_target:gk_raider
					has_leader_flag = gk_loot_receipt_from_@event_target:gk_target_planet
				}
				save_event_target_as = gk_planet_loot_receipt
			}
		}
		if = {
			limit = {
				event_target:gk_planet_loot_receipt = {
					is_variable_set = gk_pillage_damage_bank
				}
			}
			change_variable = {
				which = gk_remaining_pillage_damage_budget
				value = event_target:gk_planet_loot_receipt.gk_pillage_damage_bank
			}
		}
		event_target:gk_planet_loot_receipt = {
			kill_leader = { show_notification = no }
		}
	}

	gk_apply_planet_ransacked_surrender = yes

	# Convert the remaining devastation budget into current-rate loot for the surrender receipt
	if = {
		limit = {
			check_variable = { which = gk_remaining_pillage_damage_budget value > 0 }
		}
		set_variable = {
			which = gk_total_loot_packets
			value = gk_remaining_pillage_damage_budget
		}
		divide_variable = {
			which = gk_total_loot_packets
			value = @gk_pillage_devastation_per_packet
		}
		gk_calculate_base_loot = yes
	}

	clear_variable = gk_remaining_pillage_damage_budget
}

# Calculate base loot values (raw production * effective loot months)
# This = Planet
# Requires: gk_total_loot_packets set
# Stores: gk_loot_$RESOURCE$ on gk_storage_scope
gk_calculate_base_loot = {
	save_event_target_as = gk_loot_planet

	# Effective yield: base + modifier + war goal + pivot (all in one script value)
	set_variable = {
		which = gk_loot_months_per_packet
		value = event_target:gk_raider.value:gk_effective_pillage_yield|TARGET|event_target:gk_loot_planet.owner|
	}

	set_variable = {
		which = gk_effective_loot_months
		value = gk_total_loot_packets
	}
	multiply_variable = {
		which = gk_effective_loot_months
		value = gk_loot_months_per_packet
	}

	gk_calc_base_resource = { RESOURCE = trade }
	gk_calc_base_resource = { RESOURCE = energy }
	gk_calc_base_resource = { RESOURCE = minerals }
	gk_calc_base_resource = { RESOURCE = food }
	gk_calc_base_resource = { RESOURCE = consumer_goods }
	gk_calc_base_resource = { RESOURCE = alloys }
	gk_calc_base_resource = { RESOURCE = unity }
	gk_calc_base_resource = { RESOURCE = physics_research }
	gk_calc_base_resource = { RESOURCE = society_research }
	gk_calc_base_resource = { RESOURCE = engineering_research }
	gk_calc_base_resource = { RESOURCE = volatile_motes }
	gk_calc_base_resource = { RESOURCE = exotic_gases }
	gk_calc_base_resource = { RESOURCE = rare_crystals }
	gk_calc_base_resource = { RESOURCE = sr_living_metal }
	gk_calc_base_resource = { RESOURCE = sr_zro }
	gk_calc_base_resource = { RESOURCE = sr_dark_matter }
	gk_calc_base_resource = { RESOURCE = nanites }
	gk_add_flat_pillage_spoils = yes

	clear_variable = gk_loot_months_per_packet
	clear_variable = gk_effective_loot_months
	clear_variable = gk_total_loot_packets
}

# Calculate base loot for single resource (accumulates to storage scope)
# This = Planet
gk_calc_base_resource = {
	set_variable = {
		which = gk_temp_loot
		value = value:gk_base_loot_value|RESOURCE|$RESOURCE$|
	}
	multiply_variable = {
		which = gk_temp_loot
		value = gk_effective_loot_months
	}
	event_target:gk_storage_scope = {
		change_variable = {
			which = gk_loot_$RESOURCE$
			value = prev.gk_temp_loot
		}
	}
	clear_variable = gk_temp_loot
}

# Add non-output spoils per effective loot month. Trade is always added; mineral spoils become alloys
# on habitats and ecumenopoleis.
# This = Planet
gk_add_flat_pillage_spoils = {
	set_variable = {
		which = gk_temp_flat_spoils
		value = gk_effective_loot_months
	}
	multiply_variable = {
		which = gk_temp_flat_spoils
		value = @gk_flat_trade_spoils_per_loot_month
	}
	event_target:gk_storage_scope = {
		change_variable = {
			which = gk_flat_spoils_trade
			value = prev.gk_temp_flat_spoils
		}
	}

	set_variable = {
		which = gk_temp_flat_spoils
		value = gk_effective_loot_months
	}
	if = {
		limit = {
			is_planet_class = pc_city
		}
		multiply_variable = {
			which = gk_temp_flat_spoils
			value = @gk_flat_alloy_spoils_per_loot_month_ecu
		}
		event_target:gk_storage_scope = {
			change_variable = {
				which = gk_flat_spoils_alloys
				value = prev.gk_temp_flat_spoils
			}
		}
	}
	else_if = {
		limit = {
			is_planet_class = pc_habitat
		}
		multiply_variable = {
			which = gk_temp_flat_spoils
			value = @gk_flat_alloy_spoils_per_loot_month_habitat
		}
		event_target:gk_storage_scope = {
			change_variable = {
				which = gk_flat_spoils_alloys
				value = prev.gk_temp_flat_spoils
			}
		}
	}
	else = {
		multiply_variable = {
			which = gk_temp_flat_spoils
			value = @gk_flat_mineral_spoils_per_loot_month
		}
		event_target:gk_storage_scope = {
			change_variable = {
				which = gk_flat_spoils_minerals
				value = prev.gk_temp_flat_spoils
			}
		}
	}
	clear_variable = gk_temp_flat_spoils
}

# Apply all loot transformations (efficiency, conversions)
# This = Planet or Country
# event_target:gk_raider = Raider
# event_target:gk_storage_scope = Where loot is stored
gk_transform_loot = {
	gk_ensure_loot_variables = yes

	gk_apply_loot_conversions = yes
	gk_apply_flat_pillage_spoils = yes
}


gk_apply_loot_conversions = {
	event_target:gk_storage_scope = {
		multiply_variable = {
			which = gk_loot_trade
			value = 2
		}

		# Unity → CG at 2:1
		multiply_variable = {
			which = gk_loot_unity
			value = 0.5
		}
		change_variable = {
			which = gk_loot_consumer_goods
			value = gk_loot_unity
		}
		set_variable = { which = gk_loot_unity value = 0 }

		# Research → CG at 3:1
		multiply_variable = {
			which = gk_loot_physics_research
			value = 0.334
		}
		change_variable = {
			which = gk_loot_consumer_goods
			value = gk_loot_physics_research
		}
		set_variable = { which = gk_loot_physics_research value = 0 }

		multiply_variable = {
			which = gk_loot_society_research
			value = 0.334
		}
		change_variable = {
			which = gk_loot_consumer_goods
			value = gk_loot_society_research
		}
		set_variable = { which = gk_loot_society_research value = 0 }

		multiply_variable = {
			which = gk_loot_engineering_research
			value = 0.334
		}
		change_variable = {
			which = gk_loot_consumer_goods
			value = gk_loot_engineering_research
		}
		set_variable = { which = gk_loot_engineering_research value = 0 }
	}
}

gk_apply_flat_pillage_spoils = {
	event_target:gk_storage_scope = {
		if = {
			limit = {
				is_variable_set = gk_flat_spoils_trade
				check_variable = { which = gk_flat_spoils_trade value > 0 }
			}
			change_variable = {
				which = gk_loot_trade
				value = gk_flat_spoils_trade
			}
			clear_variable = gk_flat_spoils_trade
		}
		if = {
			limit = {
				is_variable_set = gk_flat_spoils_minerals
				check_variable = { which = gk_flat_spoils_minerals value > 0 }
			}
			change_variable = {
				which = gk_loot_minerals
				value = gk_flat_spoils_minerals
			}
			clear_variable = gk_flat_spoils_minerals
		}
		if = {
			limit = {
				is_variable_set = gk_flat_spoils_alloys
				check_variable = { which = gk_flat_spoils_alloys value > 0 }
			}
			change_variable = {
				which = gk_loot_alloys
				value = gk_flat_spoils_alloys
			}
			clear_variable = gk_flat_spoils_alloys
		}
	}
}

	gk_give_resource_to_raider = { RESOURCE = sr_dark_matter }
	gk_give_resource_to_raider = { RESOURCE = nanites }
}

gk_give_resource_to_raider = {
	event_target:gk_storage_scope = {
		if = {
			limit = {
				is_variable_set = gk_loot_$RESOURCE$
				check_variable = { which = gk_loot_$RESOURCE$ value > 0 }
			}
			event_target:gk_raider = {
				add_resource = {
					$RESOURCE$ = 1
					mult = prev.gk_loot_$RESOURCE$
				}
			}
		}
	}
}

# Send pillage notifications for immediate devastation packets
# This = Planet
# event_target:gk_raider = Raider

		create_message = {
			type = MESSAGE_TYPE_GK_PILLAGE_DEVASTATION
			localization = MESSAGE_GK_PLANET_PILLAGED
			days = 30

			variable = {
				type = name
				localization = PLANET
				scope = prev
			}

			inline_script = { script = gk_raiding/loot_message_resource_parameter RESOURCE = energy SCOPE = prev }
			inline_script = { script = gk_raiding/loot_message_resource_parameter RESOURCE = minerals SCOPE = prev }

			inline_script = { script = gk_raiding/loot_message_resource_parameter RESOURCE = consumer_goods SCOPE = prev }
			inline_script = { script = gk_raiding/loot_message_resource_parameter RESOURCE = alloys SCOPE = prev }
			inline_script = { script = gk_raiding/loot_message_resource_parameter RESOURCE = volatile_motes SCOPE = prev }
			inline_script = { script = gk_raiding/loot_message_resource_parameter RESOURCE = exotic_gases SCOPE = prev }

			inline_script = { script = gk_raiding/loot_message_resource_parameter RESOURCE = nanites SCOPE = prev }
		}
	}

	owner = {

			type = MESSAGE_TYPE_GK_OUR_PILLAGE_DEVASTATION

			days = 30
			target = prev
			variable = {
				type = name
				localization = TARGET
				scope = prev
			}
			variable = {
				type = name
				localization = FROM
				scope = event_target:gk_raider
			}
		}
	}
}

# Apply Ransacked modifier after invasion based on current devastation
# This = Planet
gk_apply_planet_ransacked_invasion = {
	if = {
		limit = { planet_devastation >= 50 }
		add_modifier = {
			modifier = gk_planet_ransacked_1
			days = @gk_surrender_aftermath_days
		}
	}
	else = {
		add_modifier = {
			modifier = gk_planet_ransacked_2
			days = @gk_surrender_aftermath_days
		}
		# Set devastation to 50
		export_trigger_value_to_variable = {
			trigger = planet_devastation
			variable = gk_dev_delta_temp
		}
		set_variable = { which = gk_dev_target_temp value = 50 }
		subtract_variable = { which = gk_dev_target_temp value = gk_dev_delta_temp }
		add_planet_devastation = gk_dev_target_temp
		clear_variable = gk_dev_delta_temp
		clear_variable = gk_dev_target_temp
	}
}

# Apply Ransacked modifier after surrender based on current devastation
# Lighter tiers when raider pivoted from pop raiding (already got value from the war)
# This = Planet
gk_apply_planet_ransacked_surrender = {
	if = {
		limit = {
			event_target:gk_raider = { has_country_flag = gk_pivoted_to_pillaging_against_@prev.owner }
		}
		# Pivoted: one tier lighter (no mod if >50, -10% if >0, -20% if 0)
		if = {
			limit = { planet_devastation < 1 }
			add_modifier = {
				modifier = gk_planet_ransacked_2
				days = @gk_surrender_aftermath_days
			}
		}
		else_if = {
			limit = { planet_devastation < 50 }
			add_modifier = {
				modifier = gk_planet_ransacked_1
				days = @gk_surrender_aftermath_days
			}
		}
	}
	else = {
		# Standard surrender tiers
		if = {
			limit = { planet_devastation < 1 }
			add_modifier = {
				modifier = gk_planet_ransacked_3
				days = @gk_surrender_aftermath_days
			}
		}
		else_if = {
			limit = { planet_devastation >= 50 }
			add_modifier = {
				modifier = gk_planet_ransacked_1
				days = @gk_surrender_aftermath_days
			}
		}
		else = {
			add_modifier = {
				modifier = gk_planet_ransacked_2
				days = @gk_surrender_aftermath_days
			}
		}
	}
}
