# GK Raiding — Privateer Enclave effects
# Event targets set by parent: gk_privateer_enclave_creator, gk_privateer_enclave_species,
#   gk_privateer_enclave_leader, gk_privateer_enclave_planet

# Assigns a personality flag to the enclave country
# This = enclave country (inside create_country effect block)
# Prefers unique personalities across existing enclaves; falls back to repeats if pool exhausted
gk_assign_privateer_personality = {
	# Automaton — exclusive to machine empires with Tactical Algorithms, always assigned
	if = {
		limit = {
			event_target:gk_privateer_enclave_creator = {
				has_civic = civic_machine_tactical_algorithms
			}
		}
		set_country_flag = gk_privateer_automaton
	}
	else = {
		random_list = {
			1 = {
				modifier = {
					factor = 0
					any_country = {
						is_country_type = gk_privateer_enclave
						has_country_flag = gk_privateer_stoic
					}
				}
				set_country_flag = gk_privateer_stoic
			}
			1 = {
				modifier = {
					factor = 0
					any_country = {
						is_country_type = gk_privateer_enclave
						has_country_flag = gk_privateer_screamer
					}
				}
				set_country_flag = gk_privateer_screamer
			}
			1 = {
				modifier = {
					factor = 0
					event_target:gk_privateer_enclave_creator = {
						NOT = { is_spiritualist = yes }
					}
				}
				modifier = {
					factor = 0
					any_country = {
						is_country_type = gk_privateer_enclave
						has_country_flag = gk_privateer_cultist
					}
				}
				set_country_flag = gk_privateer_cultist
			}
			1 = {
				modifier = {
					factor = 0
					event_target:gk_privateer_enclave_creator = {
						NOT = { is_materialist = yes }
					}
				}
				modifier = {
					factor = 0
					any_country = {
						is_country_type = gk_privateer_enclave
						has_country_flag = gk_privateer_calculating
					}
				}
				set_country_flag = gk_privateer_calculating
			}
			1 = {
				modifier = {
					factor = 0
					event_target:gk_privateer_enclave_creator = {
						NOT = { is_xenophile = yes }
					}
				}
				modifier = {
					factor = 0
					any_country = {
						is_country_type = gk_privateer_enclave
						has_country_flag = gk_privateer_merchant
					}
				}
				set_country_flag = gk_privateer_merchant
			}
		}
		# Fallback: if all eligible personalities are taken, pick without uniqueness checks
		if = {
			limit = {
				NOR = {
					has_country_flag = gk_privateer_stoic
					has_country_flag = gk_privateer_screamer
					has_country_flag = gk_privateer_cultist
					has_country_flag = gk_privateer_calculating
					has_country_flag = gk_privateer_merchant
				}
			}
			random_list = {
				1 = { set_country_flag = gk_privateer_stoic }
				1 = { set_country_flag = gk_privateer_screamer }
				1 = {
					modifier = {
						factor = 0
						event_target:gk_privateer_enclave_creator = {
							NOT = { is_spiritualist = yes }
						}
					}
					set_country_flag = gk_privateer_cultist
				}
				1 = {
					modifier = {
						factor = 0
						event_target:gk_privateer_enclave_creator = {
							NOT = { is_materialist = yes }
						}
					}
					set_country_flag = gk_privateer_calculating
				}
				1 = {
					modifier = {
						factor = 0
						event_target:gk_privateer_enclave_creator = {
							NOT = { is_xenophile = yes }
						}
					}
					set_country_flag = gk_privateer_merchant
				}
			}
		}
	}
}

# Creates the privateer enclave country
# Fleet scope | owner = original empire
# Called by create_enclave_with_country with INHERITED_ETHIC param

gk_create_privateer_enclave_country = {
	create_country = {
		name = random
		type = gk_privateer_enclave
		authority = "auth_oligarchic"
		civics = {
			civic = civic_gk_privateer_enclave
		}
		origin = "origin_default"
		species = event_target:gk_privateer_enclave_species
		flag = random
		ethos = {
			ethic = ethic_fanatic_militarist
			ethic = $INHERITED_ETHIC$
		}
		ignore_initial_colony_error = yes
		effect = {
			save_event_target_as = gk_privateer_enclave_country
			set_country_flag = gk_privateer_enclave_country
			gk_assign_privateer_personality = yes
			set_graphical_culture = event_target:gk_privateer_enclave_creator
			create_fleet = {
				settings = { spawn_debris = no }
				effect = {
					set_owner = prev
					create_ship = {
						name = random
						design = "NAME_GK_Privateer_Enclave_Station"
						graphical_culture = prev
					}
					set_location = {
						target = event_target:gk_privateer_enclave_planet
						distance = 90
					}
					save_event_target_as = gk_privateer_enclave_station
				}
			}
			set_custom_capital_location = event_target:gk_privateer_enclave_planet
			set_leader = event_target:gk_privateer_enclave_leader
			establish_communications_no_message = event_target:gk_privateer_enclave_creator
			start_situation = {
				type = gk_privateer_dividends
				target = event_target:gk_privateer_enclave_creator
			}
			# Check for active resolutions and give associated modifiers
			if = {
				limit = {
					is_active_resolution = resolution_defenseprivatization_defense_contracts
				}
				add_modifier = {
					modifier = resolution_defenseprivatization_defense_contracts_merc_naval_cap
				}
			}
			if = {
				limit = {
					is_active_resolution = resolution_defenseprivatization_private_support_troops
				}
				add_modifier = {
					modifier = resolution_defenseprivatization_private_support_troops_merc_naval_cap
				}
			}
			if = {
				limit = {
					is_active_resolution = resolution_defenseprivatization_condottieri
				}
				add_modifier = {
					modifier = resolution_defenseprivatization_condottieri_merc_naval_cap
				}
			}
			if = {
				limit = {
					is_active_resolution = resolution_defenseprivatization_security_business
				}
				add_modifier = {
					modifier = resolution_defenseprivatization_security_business_merc_naval_cap
				}
			}
			if = {
				limit = {
					is_active_resolution = resolution_defenseprivatization_corporate_warlords
				}
				add_modifier = {
					modifier = resolution_defenseprivatization_corporate_warlords_merc_naval_cap
				}
			}
			# Prevent AI from instantly hiring
			set_timed_country_flag = {
				flag = merc_ai_delay
				years = 1
			}
		}
	}
}

# Accumulate post-transform loot onto the receipt (for deferred payout)
# This = Planet (loot source)
# event_target:gk_loot_receipt = receipt leader
# event_target:gk_storage_scope = planet (temp storage with loot vars)
gk_accumulate_loot_on_receipt = {
	event_target:gk_loot_receipt = {
		gk_accumulate_resource_on_receipt = { RESOURCE = trade }
		gk_accumulate_resource_on_receipt = { RESOURCE = energy }
		gk_accumulate_resource_on_receipt = { RESOURCE = minerals }
		gk_accumulate_resource_on_receipt = { RESOURCE = food }
		gk_accumulate_resource_on_receipt = { RESOURCE = consumer_goods }
		gk_accumulate_resource_on_receipt = { RESOURCE = alloys }
		gk_accumulate_resource_on_receipt = { RESOURCE = volatile_motes }
		gk_accumulate_resource_on_receipt = { RESOURCE = exotic_gases }
		gk_accumulate_resource_on_receipt = { RESOURCE = rare_crystals }
		gk_accumulate_resource_on_receipt = { RESOURCE = sr_living_metal }
		gk_accumulate_resource_on_receipt = { RESOURCE = sr_zro }
		gk_accumulate_resource_on_receipt = { RESOURCE = sr_dark_matter }
		gk_accumulate_resource_on_receipt = { RESOURCE = nanites }
	}
}

gk_accumulate_resource_on_receipt = {
	if = {
		limit = {
			event_target:gk_storage_scope = {
				is_variable_set = gk_loot_$RESOURCE$
				check_variable = { which = gk_loot_$RESOURCE$ value > 0 }
			}
		}
		change_variable = {
			which = gk_loot_$RESOURCE$
			value = event_target:gk_storage_scope.gk_loot_$RESOURCE$
		}
	}
}

# Find station context: system, star, and patron via enclave station fleet
# This = enclave country
# Saves event_target:gk_privateer_enclave_system, gk_enclave_star, gk_privateer_patron (if patron exists)
gk_get_privateer_station_context = {
	random_owned_fleet = {
		limit = {
			any_owned_ship = {
				is_ship_size = gk_privateer_station
			}
		}
		solar_system = {
			save_event_target_as = gk_privateer_enclave_system
			star = {
				save_event_target_as = gk_enclave_star
			}
			if = {
				limit = { exists = space_owner }
				space_owner = {
					save_event_target_as = gk_privateer_patron
				}
			}
		}
	}
}

# End a privateer raid: cashout all receipts to patron, cleanup, destroy temp country
# This = enclave country
# event_target:gk_privateer_raider = temp raider country
# Expects enclave to have gk_auto_raid_ongoing flag
gk_end_privateer_raid = {
	save_event_target_as = gk_privateer_enclave_country

	# Find patron, system, and enclave star
	gk_get_privateer_station_context = yes

	# Find raid target from temp raider's flags
	if = {
		limit = { exists = event_target:gk_privateer_raider }
		random_country = {
			limit = {
				is_country_type = default
				event_target:gk_privateer_raider = {
					has_country_flag = gk_privateer_targeting_@prev
				}
			}
			save_event_target_as = gk_raid_target
			# Mark target as recently raided by this enclave
			set_timed_country_flag = {
				flag = gk_raided_by_@event_target:gk_privateer_enclave_country
				days = 720
			}
		}
	}

	# Initialize display totals on enclave scope
	gk_init_display_totals = yes
	set_variable = { which = gk_privateer_has_loot value = 0 }

	# Iterate all receipts tagged to this temp raider and cash out to patron
	if = {
		limit = {
			exists = event_target:gk_privateer_raider
			exists = event_target:gk_privateer_patron
		}
		event_target:gk_privateer_patron = {
			save_event_target_as = gk_raider
		}
		event_target:global_event_country = {
			every_owned_leader = {
				limit = {
					has_leader_flag = gk_loot_receipt_to_@event_target:gk_privateer_raider
				}
				save_event_target_as = gk_loot_receipt
				save_event_target_as = gk_storage_scope

				# Accumulate display totals before collection
				event_target:gk_privateer_enclave_country = {
					gk_accumulate_display_total = { RESOURCE = trade }
					gk_accumulate_display_total = { RESOURCE = energy }
					gk_accumulate_display_total = { RESOURCE = minerals }
					gk_accumulate_display_total = { RESOURCE = food }
					gk_accumulate_display_total = { RESOURCE = consumer_goods }
					gk_accumulate_display_total = { RESOURCE = alloys }
					gk_accumulate_display_total = { RESOURCE = volatile_motes }
					gk_accumulate_display_total = { RESOURCE = exotic_gases }
					gk_accumulate_display_total = { RESOURCE = rare_crystals }
					gk_accumulate_display_total = { RESOURCE = sr_living_metal }
					gk_accumulate_display_total = { RESOURCE = sr_zro }
					gk_accumulate_display_total = { RESOURCE = sr_dark_matter }
					gk_accumulate_display_total = { RESOURCE = nanites }
				}

				# Collect to patron
				gk_collect_loot_from_storage = yes

				# Destroy receipt
				kill_leader = { show_notification = no }
			}
		}

		# Check if we got any loot
		if = {
			limit = {
				OR = {
					check_variable = { which = gk_total_trade value > 0 }
					check_variable = { which = gk_total_energy value > 0 }
					check_variable = { which = gk_total_minerals value > 0 }
					check_variable = { which = gk_total_food value > 0 }
					check_variable = { which = gk_total_consumer_goods value > 0 }
					check_variable = { which = gk_total_alloys value > 0 }
					check_variable = { which = gk_total_volatile_motes value > 0 }
					check_variable = { which = gk_total_exotic_gases value > 0 }
					check_variable = { which = gk_total_rare_crystals value > 0 }
					check_variable = { which = gk_total_sr_living_metal value > 0 }
					check_variable = { which = gk_total_sr_zro value > 0 }
					check_variable = { which = gk_total_sr_dark_matter value > 0 }
					check_variable = { which = gk_total_nanites value > 0 }
				}
			}
			set_variable = { which = gk_privateer_has_loot value = 1 }
			gk_round_display_totals = yes
		}
	}

	# Transfer fleet back to enclave (if it still exists)
	if = {
		limit = {
			exists = event_target:gk_privateer_raider
			event_target:gk_privateer_raider = {
				any_owned_fleet = {
					num_ships > 0
				}
			}
		}
		event_target:gk_privateer_raider = {
			every_owned_fleet = {
				set_owner = event_target:gk_privateer_enclave_country
				remove_fleet_flag = gk_fleet_locked
				remove_auto_move_target = yes
				# Return to enclave station system
				if = {
					limit = { exists = event_target:gk_enclave_star }
					queue_actions = {
						move_to = event_target:gk_enclave_star
					}
				}
			}
		}
	}

	# Notify patron if loot > 0
	if = {
		limit = {
			exists = event_target:gk_privateer_patron
			check_variable = { which = gk_privateer_has_loot value > 0 }
		}
		event_target:gk_privateer_patron = {
			# Copy display vars
			set_variable = { which = gk_total_trade value = event_target:gk_privateer_enclave_country.gk_total_trade }
			set_variable = { which = gk_total_energy value = event_target:gk_privateer_enclave_country.gk_total_energy }
			set_variable = { which = gk_total_minerals value = event_target:gk_privateer_enclave_country.gk_total_minerals }
			set_variable = { which = gk_total_food value = event_target:gk_privateer_enclave_country.gk_total_food }
			set_variable = { which = gk_total_consumer_goods value = event_target:gk_privateer_enclave_country.gk_total_consumer_goods }
			set_variable = { which = gk_total_alloys value = event_target:gk_privateer_enclave_country.gk_total_alloys }
			set_variable = { which = gk_total_volatile_motes value = event_target:gk_privateer_enclave_country.gk_total_volatile_motes }
			set_variable = { which = gk_total_exotic_gases value = event_target:gk_privateer_enclave_country.gk_total_exotic_gases }
			set_variable = { which = gk_total_rare_crystals value = event_target:gk_privateer_enclave_country.gk_total_rare_crystals }
			set_variable = { which = gk_total_sr_living_metal value = event_target:gk_privateer_enclave_country.gk_total_sr_living_metal }
			set_variable = { which = gk_total_sr_zro value = event_target:gk_privateer_enclave_country.gk_total_sr_zro }
			set_variable = { which = gk_total_sr_dark_matter value = event_target:gk_privateer_enclave_country.gk_total_sr_dark_matter }
			set_variable = { which = gk_total_nanites value = event_target:gk_privateer_enclave_country.gk_total_nanites }
			country_event = { id = gk_raiding_privateers.221 }
		}
	}

	# Notify target that raiders withdrew
	if = {
		limit = {
			exists = event_target:gk_raid_target
		}
		event_target:gk_raid_target = {
			country_event = { id = gk_raiding_privateers.402 }
		}
	}

	# Cleanup flags
	remove_country_flag = gk_auto_raid_ongoing
	remove_country_flag = gk_raid_commissioned

	# Set cooldown
	set_timed_country_flag = {
		flag = gk_auto_raid_cooldown
		days = 540
	}

	# Destroy temp raider country
	if = {
		limit = {
			exists = event_target:gk_privateer_raider
		}
		event_target:gk_privateer_raider = {
			destroy_country = yes
		}
	}

	# Clear display vars
	gk_clear_display_totals = yes
}

# Clean up an active privateer raid without cashout (station destroyed, fleet lost, orphaned)
# This = enclave country (clears flags and unlocks fleets on this scope)
# Expects event_target:gk_privateer_raider to be set (temp raider country)
# Destroys all loot receipts tagged to the raider, then destroys the raider country
gk_cleanup_privateer_raid = {
	save_event_target_as = gk_privateer_enclave_country

	# Destroy all loot receipts tagged to this raider
	event_target:global_event_country = {
		every_owned_leader = {
			limit = {
				has_leader_flag = gk_loot_receipt_to_@event_target:gk_privateer_raider
			}
			kill_leader = { show_notification = no }
		}
	}

	# Destroy the temp raider country
	event_target:gk_privateer_raider = {
		every_owned_fleet = {
			delete_fleet = this
		}
		destroy_country = yes
	}

	# Clear raid flags on enclave
	remove_country_flag = gk_auto_raid_ongoing
	remove_country_flag = gk_has_active_privateer_raider
	remove_country_flag = gk_raid_commissioned

	# Unlock any lingering locked fleets
	every_owned_fleet = {
		limit = { has_fleet_flag = gk_fleet_locked }
		remove_fleet_flag = gk_fleet_locked
	}
}

# Helper: accumulate one resource from receipt to display total on enclave
# This = enclave country
gk_accumulate_display_total = {
	if = {
		limit = {
			event_target:gk_loot_receipt = {
				is_variable_set = gk_loot_$RESOURCE$
				check_variable = { which = gk_loot_$RESOURCE$ value > 0 }
			}
		}
		change_variable = {
			which = gk_total_$RESOURCE$
			value = event_target:gk_loot_receipt.gk_loot_$RESOURCE$
		}
	}
}

# Cancel an active privateer raid: no loot cashout, fleet returns, notify relevant parties
# This = enclave country
# event_target:gk_cancel_raid_initiator = the country that cancelled (patron or commissioner)
gk_cancel_privateer_raid = {
	save_event_target_as = gk_privateer_enclave_country
	gk_get_privateer_station_context = yes

	# Find the raider country
	random_country = {
		limit = {
			is_country_type = gk_privateer_raiders
			has_country_flag = gk_privateer_raider_of_@prev
		}
		save_event_target_as = gk_privateer_raider
	}

	# Find raid target from raider flags
	if = {
		limit = { exists = event_target:gk_privateer_raider }
		random_country = {
			limit = {
				is_country_type = default
				event_target:gk_privateer_raider = {
					has_country_flag = gk_privateer_targeting_@prev
				}
			}
			save_event_target_as = gk_raid_target
		}
	}

	# Clean up using existing effect (destroys receipts, debris, raider country)
	if = {
		limit = { exists = event_target:gk_privateer_raider }
		# Transfer fleet back before cleanup destroys raider
		event_target:gk_privateer_raider = {
			every_owned_fleet = {
				set_owner = event_target:gk_privateer_enclave_country
				remove_fleet_flag = gk_fleet_locked
				remove_auto_move_target = yes
				if = {
					limit = { exists = event_target:gk_enclave_star }
					queue_actions = {
						move_to = event_target:gk_enclave_star
					}
				}
			}
		}
		gk_cleanup_privateer_raid = yes
	}
	else = {
		# No raider found, just clear flags
		remove_country_flag = gk_auto_raid_ongoing
		remove_country_flag = gk_has_active_privateer_raider
		remove_country_flag = gk_raid_commissioned
		every_owned_fleet = {
			limit = { has_fleet_flag = gk_fleet_locked }
			remove_fleet_flag = gk_fleet_locked
		}
	}

	# Set cooldown
	set_timed_country_flag = {
		flag = gk_auto_raid_cooldown
		days = 360
	}

	# Notify patron if this was a commissioned raid cancelled by someone other than the patron
	if = {
		limit = {
			has_country_flag = gk_raid_commissioned
			exists = event_target:gk_privateer_patron
			exists = event_target:gk_cancel_raid_initiator
			NOT = { event_target:gk_cancel_raid_initiator = { is_same_value = event_target:gk_privateer_patron } }
		}
		event_target:gk_privateer_patron = {
			country_event = { id = gk_raiding_privateers.346 }
		}
	}

	# Notify target
	if = {
		limit = { exists = event_target:gk_raid_target }
		event_target:gk_raid_target = {
			country_event = { id = gk_raiding_privateers.347 }
		}
	}
}
