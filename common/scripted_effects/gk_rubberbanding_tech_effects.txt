####################################
# Tech rubberbanding effects      #
####################################

# Clear all tech rubberbanding modifiers
gkrb_clear_tech_rubberbanding_modifiers = {
	remove_modifier = gkrb_tech_physics_research_mod
	remove_modifier = gkrb_tech_society_research_mod
	remove_modifier = gkrb_tech_engineering_research_mod
	remove_modifier = gkrb_tech_mod
	remove_modifier = gkrb_tech_behind_mod
}

# Apply tech rubberbanding when ahead (per-resource upkeep penalties)
gkrb_apply_ahead_tech_rubberbanding_modifiers = {
	# Stage 1: Calculate adjustments for all research types
	inline_script = { script = gk_rubberbanding/apply_rubberbanding_modifier_calculate RESOURCE = physics_research TYPE = tech }
	inline_script = { script = gk_rubberbanding/apply_rubberbanding_modifier_calculate RESOURCE = society_research TYPE = tech }
	inline_script = { script = gk_rubberbanding/apply_rubberbanding_modifier_calculate RESOURCE = engineering_research TYPE = tech }
	
	# Stage 2: Apply base multiplier and modifiers (no concentration for tech)
	inline_script = { script = gk_rubberbanding/apply_rubberbanding_modifier_apply RESOURCE = physics_research TYPE = tech }
	inline_script = { script = gk_rubberbanding/apply_rubberbanding_modifier_apply RESOURCE = society_research TYPE = tech }
	inline_script = { script = gk_rubberbanding/apply_rubberbanding_modifier_apply RESOURCE = engineering_research TYPE = tech }

	# Apply marker modifier (tooltip)
	add_modifier = {
		modifier = gkrb_tech_mod
		days = -1
		multiplier = gkrb_tech_mult
	}
}

# Apply tech rubberbanding when behind (catch-up bonuses only, no per-resource modifiers)
gkrb_apply_behind_tech_rubberbanding_modifiers = {
	# Apply behind modifier with catch-up bonuses
	add_modifier = {
		modifier = gkrb_tech_behind_mod
		days = -1
		multiplier = gkrb_tech_mult
	}
}
