# Paternalism — Authoritarian golden cage
# Forces Social Welfare living standard. Stable, content, stagnant.

civic_gk_ec_paternalism = {
	playable = { always = yes }
	ai_playable = { always = yes }

	potential = {
		ethics = { NOT = { value = ethic_gestalt_consciousness } }
		authority = { NOT = { value = auth_corporate } }
	}

	possible = {
		ethics = {
			OR = {
				text = civic_tooltip_authoritarian
				value = ethic_authoritarian
				value = ethic_fanatic_authoritarian
			}
		}
	}

	random_weight = { base = @civic_default_random_weight }
	ai_weight = {
		base = @ai_civic_default_base_weight
		modifier = {
			factor = @ai_civic_personality_match_factor
			OR = {
				has_ai_personality = harmonious_hierarchy
				has_ai_personality = decadent_hierarchy
			}
		}
	}

	description = "civic_gk_ec_paternalism_effects"

	modifier = {
		planet_stability_add = 10
		pop_happiness = 0.05
		species_leader_exp_gain = -0.10
		pop_ethics_shift_speed_mult = -0.25
	}

	modification = yes
}
