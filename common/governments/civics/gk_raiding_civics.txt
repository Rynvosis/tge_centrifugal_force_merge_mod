# EMANCIPATORS     #
####################
# Egalitarian civic focused on "liberating" enslaved pops
# Grants access to Emancipation policy/war goal and bonus efficiency
civic_gk_emancipators = {
	playable = { always = yes }
	ai_playable = { always = yes }

	description = civic_gk_emancipators_effects

	potential = {
		ethics = { NOT = { value = ethic_gestalt_consciousness } }
		authority = { NOT = { value = auth_corporate } }
	}

	possible = {
		ethics = {
			OR = {
				text = civic_tooltip_egalitarian
				value = ethic_egalitarian
				value = ethic_fanatic_egalitarian
			}
			OR = {
				text = civic_tooltip_militarist
				value = ethic_militarist
				value = ethic_fanatic_militarist
			}
		}
	}

	random_weight = {
		base = 5
		modifier = {
			factor = 2
			has_ethic = ethic_fanatic_egalitarian
		}
	}

	modifier = {
		army_morale = 0.50
	}

	alternate_civic_version = civic_gk_emancipators_megacorp

	modification = yes

	# Grants access to Emancipation policy option via potential check
	# The policy itself grants the base modifiers
}

# MACHINE VARIANT            #
##############################

# Machine Intelligence despoilers — abducts pops for grid amalgamation
# Starts with organic battery pops; grants raiding CB
civic_gk_barbaric_despoilers_machine = {
	playable = { always = yes }
	ai_playable = { always = yes }

	description = civic_gk_barbaric_despoilers_machine_effects

	potential = {
		authority = {
			value = auth_machine_intelligence
		}
	}

	possible = {
		civics = {
			NOR = {
				value = civic_machine_servitor
				value = civic_machine_assimilator
				value = civic_machine_terminator
			}
		}
	}

	random_weight = {
		base = 5
		modifier = {
			factor = 0
			has_global_flag = game_started
		}
	}

	modification = no

	has_secondary_species = {
		title = civic_gk_barbaric_despoilers_machine_secondary_species
	}

	modifier = {
		planet_organic_batteries_energy_produces_add = 1
		planet_organic_batteries_unity_produces_add = 1
		councilor_gestalt_legion_exp_gain = 0.25
	}

	#TODO: Add buildable versions of the synaptic lathe buildings to this civic so they can juice up their grid amalgamated pops even better.
	#and maybe for each of the 3 ascensions, a new special building related to the ascesnion
}
