# Inline script: Apply revenue adjustment (multiply and add modifier)
# Parameters:
#   $RESOURCE$  - Resource name (e.g., energy, unity, physics_research)
#   $TYPE$      - System type (e.g., tech, trad)
#   $DIRECTION$ - upkeep or produces (resolves which modifier to apply)
#
# Expects variable gkrb_$TYPE$_$RESOURCE$_adjust to be set by the calculate pass.

# Multiply by rubberbanding multiplier
multiply_variable = {
	which = gkrb_$TYPE$_$RESOURCE$_adjust
	value = gkrb_$TYPE$_mult
}

# Round to 2 decimal places
round_variable_to_closest = {
	which = gkrb_$TYPE$_$RESOURCE$_adjust
	value = 0.01
}

# Apply modifier — scales with how far from galactic average we are
add_modifier = {
	modifier = gkrb_$TYPE$_$RESOURCE$_$DIRECTION$_mod
	days = -1
	multiplier = gkrb_$TYPE$_$RESOURCE$_adjust
}
