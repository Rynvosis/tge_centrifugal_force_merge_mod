# Centralized State - Bonus Politician Jobs (Standard, capital only)
triggered_planet_modifier = {
	potential = {
		is_capital = yes
		exists = owner
		owner = {
			OR = {
				has_valid_civic = civic_gk_ec_centralized_state
				has_valid_civic = civic_gk_ec_centralized_state_megacorp
			}
			is_worker_coop_empire = no
			is_gestalt = no
		}
	}
	job_politician_add = 1
	mult = value:gk_ec_centralized_jobs|BASE|100|
}
