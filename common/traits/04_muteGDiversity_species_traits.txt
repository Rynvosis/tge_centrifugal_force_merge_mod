 ### Carnivorous	Increased Pop Growth if Livestock on planet + Reduced if not.
 trait_mgd_carnivorous = {
	cost = 2
	category = normal
	species_possible_add = {
		can_add_genetic_traits = yes
	}
	species_possible_remove = {
		OR = {
			can_remove_beneficial_genetic_traits = yes
			can_remove_pop_growth_genetic_traits = yes
		}
	}
	opposites = { "trait_slow_breeders" }
	allowed_archetypes = { BIOLOGICAL LITHOID }
	tags = { organic positive pop_growth upkeep drawbacks }
	localized_tags = {
		SPECIES_TRAIT_MUTEGDIVERSITY_GENERIC
	}
	custom_tooltip = trait_mgd_carnivorous_effects
	triggered_planet_growth_habitability_modifier = {
		potential = {
			exists = planet
			planet = {
				AND = {
					NOT = { has_deficit = food }
					any_owned_pop_group = {
						AND = {
							is_livestock = yes
							species = {
								is_archetype = BIOLOGICAL
							}
						}
					}
				}
			}
		}
		logistic_growth_mult = 0.3
	}
	triggered_pop_group_modifier = {
		potential = {
			exists = planet
			planet = {
				NOT = {
					any_owned_pop_group = {
						is_livestock = yes
					}
				}
			}
		}
		planet_pops_food_upkeep_mult = 0.5
	}

	slave_cost = {
		energy = -500
	}
 }
