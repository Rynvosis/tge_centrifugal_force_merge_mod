trait_MST_cold_blooded = {
	cost = 2

	sorting_priority = 15

	random_weight = {
		REP = 100
	}

	allowed_planet_classes = {
		pc_desert
		pc_arid
	}
	
	allowed_planet_classes_override = {
		hidden_trigger = {
			can_add_auto_preference_traits = yes
			can_add_or_remove_phenotype_traits = yes
		}
	}

	species_possible_remove = {
		if = {
			limit = {
				exists = from
			}
			OR = {
				can_remove_beneficial_genetic_traits = yes
				can_add_or_remove_phenotype_traits = yes
			}
		}
	}
	species_potential_add = {
		OR = {
			can_do_advanced_gene_modding = yes
			can_add_or_remove_phenotype_traits = yes
		}
	}
	
	initial = yes

	allowed_archetypes = { BIOLOGICAL }
	tags = { organic positive food amenities habitability drawbacks special reptilian species }

	ai_weight = {
		weight = 1
		modifier = {
			factor = 0
			NOR = {
				has_trait = trait_pc_desert_preference
				has_trait = trait_pc_arid_preference
			}
		}
	}

	custom_tooltip_with_modifiers = trait_MST_cold_blooded_tooltip
	triggered_pop_group_modifier = {
		potential = {
			exists = planet
			planet = {
				OR = {
					is_planet_class = pc_desert
					is_planet_class = pc_arid
				}
			}
		}
		pop_amenities_usage_mult = -0.1
		planet_pops_food_upkeep_mult = -0.1
	}
	triggered_pop_group_modifier = {
		potential = {
			exists = planet
			planet = {
				NOR = {
					has_climate = dry
				}
				is_ideal = no
			}
		}
		pop_amenities_usage_mult = 0.25
	}
	# To make it be able to evaluate habitability of planet it has not settled yet. Only applies to this species
	triggered_planet_growth_habitability_modifier = {
		potential = {
			NOR = {
				has_climate = dry
			}
			is_ideal = no
		}
		pop_environment_tolerance = -0.2
	}
	modifier = {
		pc_desert_habitability = 0.2
		pc_arid_habitability = 0.2
	}
	
	opposites = {
		"trait_invasive"
	}
	
	species_class = { REP }
	localized_tags = {
		SPECIES_TRAIT_REPTILIAN
	}
	species_class_override = {
		hidden_trigger = {
			can_add_or_remove_phenotype_traits = yes
		}
	}

	icon = "gfx/interface/icons/traits/trait_MST_cold_blooded.dds"

	assembly_score = {
		modifier = {
			add = 1
			from = {
				OR = {
					is_planet_class = pc_desert
					is_planet_class = pc_arid
				}
			}
		}

		modifier = {
			subtract = 1
			from = {
				NOT = {
					has_climate = dry
				}
				is_ideal = no
			}
		}
	}
}
