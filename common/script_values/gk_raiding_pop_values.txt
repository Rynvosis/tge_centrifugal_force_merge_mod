		trigger = habitability
		trigger_scope = this

	square = yes

# Remaining bucket cap for war goal tooltip
# This/Root = attacker country, From = defender country, From.From = war
# Formula: (from_pops + bucket_taken) * rate - bucket_taken
# Reconstructs effective pool dynamically to avoid double-counting pop reduction
gk_surrender_pop_estimate = {
	base = 0
	# Start with from_pops
	complex_trigger_modifier = {
		trigger = count_owned_pop_amount
		trigger_scope = from
		mode = add
	}
	# Add bucket_pops_taken to reconstruct effective pool
	modifier = {
		from.from = { attacker = { is_same_value = root } }
		exists = event_target:gk_raid_tracker_atk_@from.from
		add = event_target:gk_raid_tracker_atk_@from.from.gk_bucket_pops_taken
	}
	modifier = {
		from.from = { defender = { is_same_value = root } }
		exists = event_target:gk_raid_tracker_def_@from.from
		add = event_target:gk_raid_tracker_def_@from.from.gk_bucket_pops_taken
	}
	# Multiply by bucket rate
	modifier = {
		from.from = { attacker = { is_same_value = root } }
		exists = event_target:gk_raid_tracker_atk_@from.from
		mult = value:gk_bucket_rate_for_tier|TRACKER|event_target:gk_raid_tracker_atk_@from.from|
	}
	modifier = {
		from.from = { defender = { is_same_value = root } }
		exists = event_target:gk_raid_tracker_def_@from.from
		mult = value:gk_bucket_rate_for_tier|TRACKER|event_target:gk_raid_tracker_def_@from.from|
	}
	modifier = {
		from.from = { attacker = { is_same_value = root } }
		NOT = { exists = event_target:gk_raid_tracker_atk_@from.from }
		mult = @gk_bucket_rate_0
	}
	modifier = {
		from.from = { defender = { is_same_value = root } }
		NOT = { exists = event_target:gk_raid_tracker_def_@from.from }
		mult = @gk_bucket_rate_0
	}
	# Subtract bucket_pops_taken
	modifier = {
		from.from = { attacker = { is_same_value = root } }
		exists = event_target:gk_raid_tracker_atk_@from.from
		subtract = event_target:gk_raid_tracker_atk_@from.from.gk_bucket_pops_taken
	}
	modifier = {
		from.from = { defender = { is_same_value = root } }
		exists = event_target:gk_raid_tracker_def_@from.from
		subtract = event_target:gk_raid_tracker_def_@from.from.gk_bucket_pops_taken
	}
	floor = yes
	min = 0
}

# Bucket Rate      #
####################

# Bucket rate derived from escalation tier — single source of truth
# Requires: event_target:gk_raid_tracker with gk_escalation_tier
# Parameterized: call as value:gk_bucket_rate_for_tier|TRACKER|event_target:gk_raid_tracker|
gk_bucket_rate_for_tier = {
	base = 0
	modifier = {
		$TRACKER$ = {
			OR = {
				NOT = { is_variable_set = gk_escalation_tier }
				check_variable = { which = gk_escalation_tier value = 0 }
			}
		}
		add = @gk_bucket_rate_0
	}
	modifier = {
		$TRACKER$ = { check_variable = { which = gk_escalation_tier value = 1 } }
		add = @gk_bucket_rate_1
	}
	modifier = {
		$TRACKER$ = { check_variable = { which = gk_escalation_tier value = 2 } }
		add = @gk_bucket_rate_2
	}
	modifier = {
		$TRACKER$ = { check_variable = { which = gk_escalation_tier value = 3 } }
		add = @gk_bucket_rate_3
	}
	modifier = {
		$TRACKER$ = { check_variable = { which = gk_escalation_tier value = 4 } }
		add = @gk_bucket_rate_4
	}
	modifier = {
		$TRACKER$ = { check_variable = { which = gk_escalation_tier value = 5 } }
		add = @gk_bucket_rate_5
	}
}

# Bucket Cap       #
####################

# Total tier quota: ceiling((pops + bucket_taken) * rate)
# For unlimited (rate 1.0): returns pops + taken (= effective pool)
# This = raidee country, event_target:gk_raid_tracker must exist
gk_tier_quota = {
	base = 0
	complex_trigger_modifier = {
		trigger = count_owned_pop_amount
		mode = add
	}
	add = event_target:gk_raid_tracker.gk_bucket_pops_taken
	# Reaping war: take all — skip rate multiplier
	modifier = {
		NOT = {
			event_target:gk_cap_raider = {
				any_war = {
					OR = {
						AND = {
							is_war_participant = { who = prev side = attackers }
							is_war_participant = { who = prevprev side = defenders }
							using_war_goal = { type = wg_gk_reaping owner = attacker }
						}
						AND = {
							is_war_participant = { who = prev side = defenders }
							is_war_participant = { who = prevprev side = attackers }
							using_war_goal = { type = wg_gk_reaping owner = defender }
						}
					}
				}
			}
		}
		mult = value:gk_bucket_rate_for_tier|TRACKER|event_target:gk_raid_tracker|
	}
	ceiling = yes
}

# Remaining bucket cap: quota - taken, min 0
# This = raidee country, event_target:gk_raid_tracker must exist
gk_bucket_remaining = {
	base = 0
	add = value:gk_tier_quota
	subtract = event_target:gk_raid_tracker.gk_bucket_pops_taken
	min = 0
}

# Auto-raid cap: (base_if_controller + stationed armies) * per_army * months
# This = country scope, event_target:gk_target_planet must exist
# Uses count_planet_army (excludes armies in ground combat)
gk_army_raid_cap = {
	base = 0
	modifier = {
		event_target:gk_target_planet = { is_controlled_by = $OWNER$ }
		add = @gk_auto_raid_base
	}
	complex_trigger_modifier = {
		trigger = count_planet_army
		trigger_scope = event_target:gk_target_planet
		parameters = {
			limit = {
				is_defensive_army = no
				owner = { is_same_value = $OWNER$ }
			}
		}
		mode = add
	}
	mult = @gk_auto_raid_per_army
	mult = $MONTHS$
}

# Abduction rate: remaining_cap / eligible_pops, clamped [0, 1]
# This = any scope with access to event_target:gk_raidee
gk_pop_abduct_rate = {
	base = 0
	add = event_target:gk_raidee.gk_remaining_cap
	divide = event_target:gk_raidee.gk_eligible_pops
	min = 0
	max = 1
}

# Proportional pops to take from a single pop group
# This = pop_group scope
# event_target:gk_raidee = country with gk_abduct_rate and gk_remaining_cap
# Returns: round(pop_group_size * rate), clamped to remaining_cap and max_per_abduction
gk_pops_to_abduct = {
	base = 0
	add = trigger:pop_group_size
	mult = value:gk_pop_abduct_rate
	ceiling = yes
	min = 0
	# Clamp to remaining cap
	max = event_target:gk_raidee.gk_remaining_cap
}

# Tribute Base     #
####################

# Tribute base amount: max(primary_income, alloys_income*4) * 12 * years
# Uses abs trick: max(a,b) = (|a-b| + a + b) / 2
# This = raider, event_target:gk_raidee = target, event_target:gk_raid_tracker = tracker
# $PRIMARY$ = energy or food (caller resolves bio-ship)
gk_tribute_base_value = {
	base = 0
	# a - b
	add = event_target:gk_raidee.value:income_mult|RESOURCE|$PRIMARY$|MULT|1|
	subtract = event_target:gk_raidee.value:income_mult|RESOURCE|alloys|MULT|4|
	abs = yes
	# + a + b
	add = event_target:gk_raidee.value:income_mult|RESOURCE|$PRIMARY$|MULT|1|
	add = event_target:gk_raidee.value:income_mult|RESOURCE|alloys|MULT|4|
	divide = 2
	# Scale by production years
	multiply = 12
	multiply = value:gk_tribute_production_years
	# Floor (50/month * 12 = 600 min) and round
	min = 600
	round_to = 100
}

# Dynamic influence cost: 50 + 25 * tier * (yield / raider_pops) / 0.05, round to 25
# This = raider, From = raidee, event_target:gk_raid_tracker must exist
gk_escalation_influence_cost = {
	base = 0
	# Start with bucket yield (raidee pops * next tier rate)
	add = value:gk_display_bucket_remaining_value
	# Divide by raider pops to get ratio
	divide = value:gk_raider_pop_count
	# Divide by baseline (0.05) to get significance factor
	divide = @gk_significant_raid_baseline
	# Multiply by cost per tier
	multiply = @gk_escalation_cost_per_tier
	# Multiply by current tier
	multiply = event_target:gk_raid_tracker.gk_escalation_tier
	# Round to nearest 25
	round_to = 25
	min = 0
	# Add base cost
	add = @gk_escalation_base_cost
	# Double cost for opportunistic raiding (no explicit raiding war goal)
	modifier = {
		from = { has_country_flag = gk_opportunistic_raid_authorized_by_@root }
		NOT = { gk_is_plunder_war_with_from = yes }
		NOT = { gk_is_emancipation_war_with_from = yes }
		mult = @gk_opportunistic_cost_multiplier
	}
}
