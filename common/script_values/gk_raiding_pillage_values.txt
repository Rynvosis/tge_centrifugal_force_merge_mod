# Effective pillage yield: base + modifier + war goal bonus if applicable
# This = raider country
# Parameter: TARGET = target country scope to check war goal against
gk_effective_pillage_yield = {
	base = @gk_base_loot_months_per_packet
	add = modifier:gk_pillage_yield
	# +3 if in pillage war against this target
	modifier = {
		any_war = {
			OR = {
				AND = {
					is_war_participant = { who = prev side = attackers }
					is_war_participant = { who = $TARGET$ side = defenders }
					using_war_goal = { type = wg_plunder_raid owner = attacker }
				}
				AND = {
					is_war_participant = { who = prev side = defenders }
					is_war_participant = { who = $TARGET$ side = attackers }
					using_war_goal = { type = wg_plunder_raid owner = defender }
				}
			}
		}
		add = @gk_pillage_war_bonus_loot_months
	}
	# -2 if pivoted from pop raiding (offsets war goal +3, nets +1)
	modifier = {
		has_country_flag = gk_pivoted_to_pillaging_against_@$TARGET$
		add = -2
	}
}

# Remaining devastation budget for loot extraction
# This = Planet, event_target:gk_loot_receipt = receipt with damage bank
# $MAX$ = max devastation to consider (100 for invasion, 50 for surrender)
gk_remaining_pillage_budget = {
	base = $MAX$
	subtract = trigger:planet_devastation
	modifier = {
		exists = event_target:gk_loot_receipt
		event_target:gk_loot_receipt = { is_variable_set = gk_pillage_damage_bank }
		add = event_target:gk_loot_receipt.gk_pillage_damage_bank
	}
	min = 0
}

gk_base_loot_value = {
	base = 0
	complex_trigger_modifier = {
		trigger = planet_resource_compare
		parameters = {
			resource = $RESOURCE$
			type = produces
		}
		mode = add
	}
}
