# Conditional own-contribution line: only shows when own < total
defined_text = {
	name = GkOwnContributionLine
	text = {
		trigger = {
			exists = event_target:gk_own_contributor
			exists = event_target:gk_raid_tracker
			NOT = {
				check_variable = {
					which = event_target:gk_own_contributor.gk_contributor_pops_taken
					value >= event_target:gk_raid_tracker.gk_pops_taken
				}
			}
		}
		localization_key = GK_OWN_CONTRIBUTION_LINE
	}
	text = {
		trigger = {
			OR = {
				NOT = { exists = event_target:gk_own_contributor }
				NOT = { exists = event_target:gk_raid_tracker }
				check_variable = {
					which = event_target:gk_own_contributor.gk_contributor_pops_taken
					value >= event_target:gk_raid_tracker.gk_pops_taken
				}
			}
		}
		localization_key = GK_EMPTY_STRING
	}
}

# Conditional occupied pops line for tooltips
defined_text = {
	name = GkOccupiedPopsLine
	text = {
		trigger = {
			check_variable = {
				which = value:gk_occupied_eligible_pops_value
				value > 0
			}
		}
		localization_key = GK_OCCUPIED_POPS_LINE
	}
	text = {
		trigger = {
			NOT = {
				check_variable = {
					which = value:gk_occupied_eligible_pops_value
					value > 0
				}
			}
		}
		localization_key = GK_EMPTY_STRING
	}
}

# Dynamic press tooltip: normal vs final tier
defined_text = {
	name = GkPressTooltip
	text = {
		trigger = {
			exists = event_target:gk_raid_tracker
			event_target:gk_raid_tracker = {
				check_variable = { which = gk_escalation_tier value >= 4 }
			}
			gk_is_plunder_war_with_from = yes
		}
		localization_key = GK_RAIDING_ESCALATION_PRESS_FINAL_TOOLTIP
	}
	text = {
		trigger = {
			exists = event_target:gk_raid_tracker
			event_target:gk_raid_tracker = {
				check_variable = { which = gk_escalation_tier value >= 4 }
			}
			gk_is_emancipation_war_with_from = yes
		}
		localization_key = GK_RAIDING_ESCALATION_PRESS_FINAL_EMANCIPATION_TOOLTIP
	}
	text = {
		trigger = {
			OR = {
				NOT = { exists = event_target:gk_raid_tracker }
				event_target:gk_raid_tracker = {
					NOT = { check_variable = { which = gk_escalation_tier value >= 4 } }
				}
			}
		}
		localization_key = GK_RAIDING_ESCALATION_PRESS_NORMAL_TOOLTIP
	}
}
