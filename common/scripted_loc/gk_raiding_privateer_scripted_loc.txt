# Overwrite vanilla GetMercenaryLoc to add privateer branch
defined_text = {
	name = GetMercenaryLoc
	random = no
	text = {
		trigger = { has_country_flag = mindwarden_diplomacy }
		localization_key = MINDWARDEN
	}
	text = {
		trigger = { has_country_flag = gk_privateer_diplomacy }
		localization_key = GK_PRIVATEER_LOC
	}
	default = MERCENARY
}
