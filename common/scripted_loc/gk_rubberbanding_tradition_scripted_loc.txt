}

# Unity upkeep row — shown in ahead modifier tooltip
defined_text = {
	name = GKRBTradUnityUpkeepRow
	text = {
		trigger = {
			check_variable = {
				which = gkrb_trad_unity_adjust
				value >= 0.1
			}
		}
		localization_key = "GKRB_ROW_TRAD_UNITY_UPKEEP"
	}
	text = {
		trigger = {
			check_variable = {
				which = gkrb_trad_unity_adjust
				value < 0.1
			}
		}
		localization_key = "GKRB_EMPTY"
	}
}
