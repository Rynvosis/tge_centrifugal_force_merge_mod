# ----- Ahead (upkeep) rows -----
defined_text = {
	name = GKRBTechPhysicsResearchUpkeepRow
	text = {
		trigger = {
			check_variable = {
				which = gkrb_tech_physics_research_adjust
				value >= 0.1
			}
		}
		localization_key = "GKRB_ROW_TECH_PHYSICS_RESEARCH_UPKEEP"
	}
	text = {
		trigger = {
			check_variable = {
				which = gkrb_tech_physics_research_adjust
				value < 0.1
			}
		}
		localization_key = "GKRB_EMPTY"
	}
}
