situation_breach_shroud = {

	desc = situation_breach_shroud_desc

			owner = { is_hive_empire = yes }

			}
		}
		picture = GFX_evt_machine_sapience
	}
	picture = {

			owner = {

		owner = {
			set_country_flag = can_access_shroud

				set_situation_flag = breach_shroud_stage_1_started

			end = @breach_shroud_stage_2_end

					id = shroud.2760

			icon = GFX_situation_stage_3
			icon_background = GFX_situation_stage_frame_green
			end = @breach_shroud_end_value
			custom_tooltip = situation_breach_shroud_effects

				situation_event = {
					id = shroud.2780

			custom_tooltip = approach_situation_beneficial_tooltip

			country_unity_produces_mult = -0.25

			}
		}
		on_select = {
			custom_tooltip = breach_shroud_commune_resources_tooltip
			if = {
				limit = {
					owner = { NOT = { has_country_flag = zro_found } }

				custom_tooltip = breach_shroud_commune_resources_unknown_tt

			country_unity_produces_mult = -0.50

		icon_background = GFX_situation_approach_bg_yellow

			owner = { has_technology = tech_mine_zro }

				trigger = {

						AND = {
							current_stage = stage_2
							owner = {
								NOT = {

							current_stage = stage_3

				custom_tooltip = breach_shroud_commune_planes_discovered_tt

			country_unity_produces_mult = -0.50

		potential = {
			has_astral_planes_dlc = yes
			owner = { has_technology = tech_astral_harvesting }
		}
		on_select = {
			set_situation_flag = chaotic_approach
			custom_tooltip = approach_situation_chaotic_tooltip

			country_unity_produces_mult = 0.1

			base = 0
			modifier = {
				add = 5 # Higher priority than ethic choices, but zro tech is still higher prio

			modifier = {
				factor = 10
				owner = { has_resource = { type = astral_threads amount > 100 } }
