gk_ec_collectivist = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_collectivist
	unique = yes
	support_multiplier = 1
	visual_ethic = "gk_ec_collectivist_visual_ethic"

	resources = {
		category = pop_factions
		inline_script = pop_faction_types/shared_unity_production
		produces = {
			trigger = {
				owner = {
					OR = {
						has_civic = civic_crowdsourcing
						has_civic = civic_corporate_crowdsourcing
					}
				}
			}
			society_research = @faction_major_output
			engineering_research = @faction_minor_output
			mult = @crowdsourcing_multiplier
		}
	}

	extortion_resource = {
		category = pop_factions
		produces = {
			trigger = {
				owner = {
					NOR = {
						has_civic = civic_crowdsourcing
						has_civic = civic_corporate_crowdsourcing
					}
				}
			}
			energy = @faction_base_output
			mult = @faction_extortion_energy_ratio
		}
		produces = {
			trigger = {
				owner = {
					OR = {
						has_civic = civic_crowdsourcing
						has_civic = civic_corporate_crowdsourcing
					}
				}
			}
			society_research = @faction_major_output
			engineering_research = @faction_minor_output
			mult = @faction_extortion_crowdsourcing_ratio
		}
	}

	is_potential = {
		is_faction_allowed = yes
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		is_shackled_robot = no
		can_think = yes
	}

	attraction = {
		base = 150
		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { is_ruler = yes }
			}
		}
		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
		modifier = {
			factor = 0.25
			NOT = { is_same_species = owner_species }
		}
		modifier = {
			factor = 0
			planet = { has_modifier = planet_culture_shock }
		}
	}

	leader = {
		base = 100
		modifier = {
			factor = 0
			is_event_leader = yes
		}
	}

	# GRADUATED DEMAND: WELFARE (3 tiers — by living standard quality)
	# Tier 1: No welfare standards at all — always red
	demand = {
		title = "GK_EC_COLLECTIVIST_WELFARE"
		unfulfilled_title = "GK_EC_COLLECTIVIST_NO_WELFARE"
		desc = "GK_EC_COLLECTIVIST_WELFARE_DESC"

		fulfilled_effect = 0.001
		unfulfilled_effect = -10

		potential = {
			exists = owner
			owner = { NOT = { gk_ec_has_any_welfare = yes } }
		}

		trigger = {
			always = no
		}
	}

	# Tier 2: Good standards but no utopian/shared_burden
	demand = {
		title = "GK_EC_COLLECTIVIST_WELFARE"
		unfulfilled_title = "GK_EC_COLLECTIVIST_NO_WELFARE"
		desc = "GK_EC_COLLECTIVIST_WELFARE_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				gk_ec_has_any_welfare = yes
				NOT = { gk_ec_has_ideal_welfare = yes }
			}
		}

		trigger = {
			always = yes
		}
	}

	# Tier 3: Ideal welfare (utopian or shared_burden)
	demand = {
		title = "GK_EC_COLLECTIVIST_WELFARE"
		unfulfilled_title = "GK_EC_COLLECTIVIST_NO_WELFARE"
		desc = "GK_EC_COLLECTIVIST_WELFARE_DESC"

		unfulfilled_effect = -0.001

		potential = {
			exists = owner

		}

		trigger = {
			always = yes
		}
	}

		fulfilled_effect = 0.001
		unfulfilled_effect = -5

		potential = {

	# Tier 1: Neither worker coop NOR shared_burden/worker_ownership LS — always red
	demand = {
		title = "GK_EC_COLLECTIVIST_WORKER_OWNERSHIP"
		unfulfilled_title = "GK_EC_COLLECTIVIST_NO_WORKER_OWNERSHIP"

		fulfilled_effect = 0.001
		unfulfilled_effect = -10

			}
		}

		trigger = {
			always = no
		}
	}

	# Tier 2: One condition met (worker coop OR shared_burden/worker_ownership, not both)
	demand = {
		title = "GK_EC_COLLECTIVIST_WORKER_OWNERSHIP"
		unfulfilled_title = "GK_EC_COLLECTIVIST_NO_WORKER_OWNERSHIP"
		desc = "GK_EC_COLLECTIVIST_WORKER_OWNERSHIP_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				OR = {
					AND = {
						is_worker_coop_empire = yes
						NOT = { gk_ec_has_worker_ownership_ls = yes }
					}
					AND = {
						is_worker_coop_empire = no
						gk_ec_has_worker_ownership_ls = yes
					}
				}
			}
		}

		trigger = {
			always = yes
		}
	}
