gk_ec_individualist = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_individualist
	unique = yes
	support_multiplier = 1
	visual_ethic = "gk_ec_individualist_visual_ethic"

	resources = {
		category = pop_factions
		inline_script = pop_faction_types/shared_unity_production
		produces = {
			trigger = {
				owner = {
					OR = {
						has_civic = civic_crowdsourcing
						has_civic = civic_corporate_crowdsourcing
					}
				}
			}
			society_research = @faction_major_output
			engineering_research = @faction_minor_output
			mult = @crowdsourcing_multiplier
		}
	}

	extortion_resource = {
		category = pop_factions
		produces = {
			trigger = {
				owner = {
					NOR = {
						has_civic = civic_crowdsourcing
						has_civic = civic_corporate_crowdsourcing
					}
				}
			}
			energy = @faction_base_output
			mult = @faction_extortion_energy_ratio
		}
		produces = {
			trigger = {
				owner = {
					OR = {
						has_civic = civic_crowdsourcing
						has_civic = civic_corporate_crowdsourcing
					}
				}
			}
			society_research = @faction_major_output
			engineering_research = @faction_minor_output
			mult = @faction_extortion_crowdsourcing_ratio
		}
	}

	is_potential = {
		is_faction_allowed = yes
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		is_shackled_robot = no
		can_think = yes
	}

	attraction = {
		base = 150
		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { is_ruler = yes }
			}
		}
		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
		modifier = {
			factor = 0.25
			NOT = { is_same_species = owner_species }
		}
		modifier = {
			factor = 0
			planet = { has_modifier = planet_culture_shock }
		}
	}

	leader = {
		base = 100
		modifier = {
			factor = 0
			is_event_leader = yes
		}
	}

	# GRADUATED DEMAND: COMMERCIAL_PACTS (3 tiers — mutually exclusive potentials)
	# Tier 1: 0-1 commercial pacts — always red
	demand = {
		title = "GK_EC_INDIVIDUALIST_COMMERCIAL_PACTS"
		unfulfilled_title = "GK_EC_INDIVIDUALIST_NO_COMMERCIAL_PACTS"
		desc = "GK_EC_INDIVIDUALIST_COMMERCIAL_PACTS_DESC"

		fulfilled_effect = 0.001
		unfulfilled_effect = -5

		potential = {
			exists = owner
			owner = {
				num_communications >= 2
				NOT = {
					count_relation = {
						limit = { has_commercial_pact = root.owner }
						count >= 2
					}
				}
			}
		}

		trigger = {
			always = no
		}
	}

	# Tier 2: 2-3 commercial pacts
	demand = {
		title = "GK_EC_INDIVIDUALIST_COMMERCIAL_PACTS"
		unfulfilled_title = "GK_EC_INDIVIDUALIST_NO_COMMERCIAL_PACTS"
		desc = "GK_EC_INDIVIDUALIST_COMMERCIAL_PACTS_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				num_communications >= 2
				count_relation = {
					limit = { has_commercial_pact = root.owner }
					count >= 2
				}
				NOT = {
					count_relation = {
						limit = { has_commercial_pact = root.owner }
						count >= 4
					}
				}
			}
		}

		trigger = {
			always = yes
		}
	}

	# Tier 3: 4+ commercial pacts

		title = "GK_EC_INDIVIDUALIST_COMMERCIAL_PACTS"

		fulfilled_effect = 15
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {

		if = {

			remove_modifier = gk_ec_suppressed_individualist
		}
	}

			title = "EMBRACE_FACTION"
			description = "EMBRACE_FACTION_DESC"
			cost = {
				unity = 5000
			}
			potential = {
				exists = owner
				owner = {
					OR = {
						is_subject = no
						NOT = { any_agreement = { agreement_preset = preset_dominion } }
					}
				}
			}
			valid = {
				custom_tooltip = {
					fail_text = EMBRACE_FACTION_COOLDOWN

						has_ethic = "ethic_fanatic_individualist"

			}
			effect = {

					}

				hidden_effect = {

					parameter:empire = {
						every_relation = {
							limit = {
								is_ai = no
								is_country_type = default

									who = prev

					factor = 0

							count < 4
